18template <class FX>
19class FewPointsCache {
20public:
21 FewPointsCache() = default;
22 FewPointsCache(array<XYZ, 1> points_)
23 : points(std::move(points_)),
24 hashes(points.size()),
25 cache(points.size()) {
26 auto n_points = points.size();
27 array<double, 1> ps(n_points);
28 bool is_good = false;
29
30 auto random_projections = std::vector<XYZ>{
31 XYZ{0.4812687223074247, 0.4901126875380488, 0.748396406012264},
32 XYZ{0.5893000897592624, 0.2328024062253076, 0.9678789902593754},
33 XYZ{0.312442457165149, 0.3106755425491958, 0.9489204175487449},
34 XYZ{0.637790969749397, 0.0754455129073677, 0.3403685854340376},
35 XYZ{0.5338831876857503, 0.9213227083629082, 0.5431566971029622},
36 XYZ{0.2385209039087733, 0.913388571618219, 0.2053846497098911},
37 XYZ{0.2996192337558674, 0.6389134569121155, 0.7896639672566169},
38 XYZ{0.3261298736535007, 0.7045295132395185, 0.642961279569149},
39 XYZ{0.0931963937441646, 0.1815748333117181, 0.7493070549814564},
40 XYZ{0.5116768712958907, 0.9587701460773556, 0.6998062953160484}};
41
42 for (const auto &rp : random_projections) {
43 for (int_t i = 0; i < n_points; ++i) {
44 ps[i] = zisa::dot(rp, points[i]);
45 }
46
47 std::sort(ps.begin(), ps.end());
48
49 double dx_max = ps[n_points - 1] - ps[0];
50 double dx_min
51 = zisa::reduce::min(serial_policy{},
52 index_range(0, n_points - 1),
53 [&ps](int_t i) { return ps[i + 1] - ps[i]; });
54
55 // Are the points are reasonably well separated?
56 if (dx_min > dx_max / 100000.0) {
57 atol = 1e-10 * dx_max;
58 projection = rp;
59 is_good = true;
60 break;
61 }
62 }
63
64 LOG_ERR_IF(!is_good, "Failed to find a suitable projection.");
65
66 std::sort(points.begin(), points.end(), [this](const XYZ &x, const XYZ &y) {
67 return zisa::dot(projection, x) < zisa::dot(projection, y);
68 });
69
70 for (int_t i = 0; i < n_points; ++i) {
71 hashes[i] = zisa::dot(projection, points[i]);
72 }
73 }
74
75 void update(const std::function<FX(const XYZ &x)> &f_) {
76 this->f = f_;
77
78 auto n_points = points.size();
79 for (int_t i = 0; i < n_points; ++i) {
80 cache[i] = f(points[i]);
81 }
82 }
83
84 FX get(const XYZ &x) const {
85 double hash = zisa::dot(projection, x);
86 auto it = std::lower_bound(hashes.cbegin(), hashes.cend(), hash - atol);
87 if (std::abs(*it - hash) <= atol) {
88 return cache[integer_cast<int_t>(it - hashes.begin())];
89 } else {
90 return f(x);
91 }
92 }
93
94private:
95 std::function<FX(const XYZ &x)> f;
96 array<XYZ, 1> points;
97 array<double, 1> hashes;
98 array<FX, 1> cache;
99
100 XYZ projection;
101 double atol;
102};
103
104}
