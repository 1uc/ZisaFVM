70 void write_grid() override {
71 auto grid = this->choose_grid();
72 auto fng = this->choose_file_name_generator();
73
74 if (is_gathered_visualization()) {
75 auto vis_info = choose_gathered_vis_info();
76 auto gatherer_factory = choose_gatherer_factory();
77 auto gatherer = gatherer_factory->template create_object<CellFlags, 1>();
78
79 if (vis_info->h5_comm != MPI_COMM_NULL) {
80
81 auto cell_flags = array<CellFlags, 1>(vis_info->n_vis_cells());
82 gatherer.copy_local_patch(
83 array_view(cell_flags),
84 const_slice(array_const_view(grid->cell_flags),
85 0,
86 vis_info->n_local_cells));
87
88 gatherer.receive(array_view(cell_flags));
89 apply_permutation(array_view(cell_flags), *vis_info->permutation);
90
91 auto flags_name = std::filesystem::path(fng->dirname) / "cell_flags.h5";
92 auto file_dims = choose_gathered_file_info();
93 auto writer = HDF5UnstructuredWriter(
94 flags_name, file_dims, HDF5Access::overwrite);
95
96 save(writer, cell_flags, "cell_flags");
97 } else {
98 gatherer.send(const_slice(
99 array_const_view(grid->cell_flags), 0, vis_info->n_local_cells));
100 }
101
102 } else if (is_split_visualization()) {
103 auto dir = string_format("part-%04d/", mpi_rank);
104 create_directory(dir);
105
106 auto writer = HDF5SerialWriter(dir + "cell_flags.h5");
107 save(writer, grid->cell_flags, "cell_flags");
108
109 } else if (is_unstructured_visualization()) {
110 auto flags_name = std::filesystem::path(fng->dirname) / "cell_flags.h5";
111
112 auto file_dims = choose_file_dimensions();
113 auto writer = HDF5UnstructuredWriter(
114 flags_name, file_dims, HDF5Access::overwrite);
115
116 save(writer, grid->cell_flags, "cell_flags");
117 } else {
118 LOG_ERR("Broken logic.");
119 }
120 }

128 void do_post_run(const std::shared_ptr<AllVariables> &u1) override {
129 if (!has_key(this->params, "reference")) {
130 // Post processing the reference solution is not requested.
131 return;
132 }
133
134 auto u1_ref = this->deduce_reference_solution(*u1);
135 down_sample(u1_ref, "reference.h5");
136
137 auto fng = this->choose_file_name_generator();
138 auto steady_state_filename = fng->steady_state_filename;
139 auto u_delta = load_all_vars(steady_state_filename);
140
141 auto halo_exchange = choose_halo_exchange();
142 (*halo_exchange)(*u_delta);
143 (*halo_exchange).wait();
144
145 for (int_t i = 0; i < u_delta->size(); ++i) {
146 (*u_delta)[i] = (*u1)[i] - (*u_delta)[i];
147 }
148
149 auto grid = this->choose_grid();
150 auto local_eos = this->choose_local_eos();
151 auto weno_params = this->choose_weno_reference_params();
152 auto local_rc_params = this->choose_local_rc_params();
153 auto rc = make_reconstruction_array<NoEquilibrium,
154 CWENO_AO,
155 UnityScaling,
156 typename super::eos_t,
157 typename super::gravity_t>(
158 grid, weno_params, *local_eos, this->gravity, local_rc_params);
159 auto grc = std::make_shared<
160 EulerGlobalReconstruction<NoEquilibrium, CWENO_AO, UnityScaling>>(
161 weno_params, std::move(rc));
162
163 auto u_delta_ref = this->deduce_reference_solution_eq(*u_delta, grc);
164 down_sample(u_delta_ref, "delta.h5");
165 }

167 std::shared_ptr<AllVariables> load_all_vars(const std::string &filename) {
168 if (is_gathered_visualization()) {
169 auto grid = this->choose_grid();
170 auto vis_info = choose_gathered_vis_info();
171 auto scatter_tag = ZISA_MPI_TAG_LOAD_ALL_VARS;
172 if (vis_info->h5_comm != MPI_COMM_NULL) {
173 auto file_dims = choose_gathered_file_info();
174 auto reader = HDF5UnstructuredReader(filename, file_dims);
175
176 auto all_vars_patch
177 = AllVariables::load(reader, all_labels<euler_var_t>());
178
179 reverse_permutation(array_view(all_vars_patch.cvars),
180 *vis_info->permutation);
181
182 auto darray_info
183 = make_distributed_array_info(vis_info->vis_boundaries);
184
185 auto cvars_scatterer
186 = std::make_unique<MPISingleNodeArrayScatterer<double, 2>>(
187 darray_info, vis_info->vis_comm, scatter_tag);
188
189 auto avars_scatterer
190 = std::make_unique<MPISingleNodeArrayScatterer<double, 2>>(
191 darray_info, vis_info->vis_comm, scatter_tag + 1);
192
193 auto scatterer = AllVariablesScatterer(std::move(cvars_scatterer),
194 std::move(avars_scatterer),
195 vis_info->n_local_cells,
196 grid->n_cells);
197
198 return std::make_shared<AllVariables>(
199 scatterer.scatter(all_vars_patch));
200
201 } else {
202 auto cvars_scatterer
203 = std::make_unique<MPISingleNodeArrayScatterer<double, 2>>(
204 nullptr, vis_info->vis_comm, scatter_tag);
205
206 auto avars_scatterer
207 = std::make_unique<MPISingleNodeArrayScatterer<double, 2>>(
208 nullptr, vis_info->vis_comm, scatter_tag + 1);
209
210 auto scatterer = AllVariablesScatterer(std::move(cvars_scatterer),
211 std::move(avars_scatterer),
212 vis_info->n_local_cells,
213 grid->n_cells);
214
215 return std::make_shared<AllVariables>(scatterer.scatter(5, 0));
216 }
217 }
218
219 if (is_unstructured_visualization()) {
220 auto file_dims = choose_file_dimensions();
221 auto reader = HDF5UnstructuredReader(filename, file_dims);
222
223 return std::make_shared<AllVariables>(
224 AllVariables::load(reader, all_labels<euler_var_t>()));
225 }
226
227 LOG_ERR("Implement missing case.");
228 }

230 void down_sample(const std::shared_ptr<ReferenceSolution> &ref_soln,
231 const std::string &filename) {
232
233 std::vector<std::string> coarse_grid_paths
234 = this->params["reference"]["coarse_grids"];
235
236 auto serialize = [this](const std::string &filename,
237 MPI_Comm small_comm,
238 const DistributedGrid &small_dgrid,
239 const AllVariables &all_vars) {
240 auto vis_info = this->compute_gathered_vis_info(small_comm, small_dgrid);
241 auto gatherer_factory = this->compute_gatherer_factory(*vis_info);
242
243 auto all_var_dims = this->choose_all_variable_dims();
244 auto simulation_clock = this->choose_simulation_clock();
245
246 auto fng = std::make_shared<SingleFileNameGenerator>(filename);
247 auto file_dims = compute_gathered_file_info(*vis_info);
248 auto local_eos = this->compute_local_eos(file_dims->n_cells_local);
249 auto dump_snapshot
250 = std::make_shared<ParallelDumpSnapshot<typename super::eos_t>>(
251 local_eos, fng, file_dims);
252
253 auto vis = make_gathered_visualization(std::move(vis_info),
254 std::move(gatherer_factory),
255 std::move(dump_snapshot),
256 all_var_dims);
257
258 (*vis)(all_vars, *simulation_clock);
259 };
260
261 auto fine_grid = this->choose_grid();
262 auto mask = super::boundary_mask();
263
264 auto interpolation = [&ref_soln](int_t i, const XYZ &x, int_t k) {
265 return ref_soln->q_ref(x, k, i);
266 };
267
268 auto dref = DistributedReferenceSolution(
269 serialize, fine_grid, interpolation, euler_var_t::size());
270
271 for (const auto &coarse_base_folder : coarse_grid_paths) {
272 auto coarse_folder
273 = std::filesystem::path(coarse_base_folder) / "partitioned";
274 dref.compute_and_save(
275 filename, coarse_folder, small_comm_size(coarse_folder), mask);
276 }
277 }
278
279 int small_comm_size(const std::string &coarse_folder) {
280 auto sizes = std::vector<int>();
281 for (auto &p : std::filesystem::directory_iterator(coarse_folder)) {
282 if (p.is_directory()) {
283 auto b = p.path().filename().string();
284 auto s = stoi(b);
285 if (string_format("%d", s) == b) {
286 if (s <= mpi_comm_size) {
287 sizes.push_back(s);
288 }
289 }
290 }
291 }
292
293 LOG_ERR_IF(sizes.empty(), "Failed to deduce any valid `small_comm_size`.");
294 return *std::max_element(sizes.begin(), sizes.end());
295 }

317 std::shared_ptr<Grid> compute_grid() const override {
318 // 1. Load in a oversized chunk.
319 // 2. Decide how much we really need based on the stencil.
320 // 3. Generate correctly sized grid.
321
322 // Side effect: - store the stencils.
323 // - store distributed grid info.
324
325 auto stencil_params = this->choose_stencil_params();
326 auto qr_degrees = this->choose_qr_degrees();
327 auto [stencils, dgrid, grid] = zisa::load_local_grid(subgrid_name(),
328 super::boundary_mask(),
329 stencil_params,
330 qr_degrees,
331 mpi_rank);
332
333 this->stencils_ = stencils;
334 this->distributed_grid_ = dgrid;
335 this->enforce_cell_flags(*grid);
336
337 {
338 auto p = this->distributed_grid_->partition;
339 auto k = zisa::mpi::rank();
340
341 auto n_local_cells = std::count_if(p.begin(), p.end(), [k](int_t i) {
342 return i == integer_cast<int_t>(k);
343 });
344
345 auto n_non_local_cells = p.size() - n_local_cells;
346
347 auto lc = string_format("local_cells-%d.txt", k);
348 write_string_to_file(string_format("%d", n_local_cells), lc);
349
350 auto nlc = string_format("non_local_cells-%d.txt", k);
351 write_string_to_file(string_format("%d", n_non_local_cells), nlc);
352 }
353
354 return grid;
355 }

357 std::function<bool(const Grid &, int_t i)> boundary_mask() const override {
358 auto physical_mask = super::boundary_mask();
359 auto dgrid = choose_distributed_grid();
360 return [physical_mask, dgrid, mpi_rank = this->mpi_rank](const Grid &grid,
361 int_t i) {
362 return dgrid->partition[i] != integer_cast<int_t>(mpi_rank)
363 || physical_mask(grid, i);
364 };
