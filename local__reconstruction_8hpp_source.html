45 
46  auto n_points = grid->cells[i_cell].qr.points.size();

69  void compute_equilibrium(const array_view<cvars_t, 1> &u_local) {
70  const auto &u0 = u_local(int_t(0));
71  auto rhoE_self = RhoE{u0[0], internal_energy(u0)};
72 
73  auto &l2g = rc.local2global();
74  scale = scaling(rhoE_self);
75  eq.solve(rhoE_self, grid->cells(i_cell));
76 
77  point_values_cache.update(
78  [this](const XYZ &x) { return eq.extrapolate_full(x); });
79 
80  for (int_t il = 0; il < l2g.size(); ++il) {
81  rhoEbar_cache(il) = eq.extrapolate(grid->cells(l2g[il]));
82  }
83 
84  steps_since_recompute = 0;
85  }
86 
87  void recompute_equilibrium(const array_view<cvars_t, 1> &u_local) {
88  if (steps_since_recompute % steps_per_recompute == 0) {
89  compute_equilibrium(u_local);
90  } else {
91  const auto &u0 = u_local(int_t(0));
92  auto [rho, E] = RhoE{u0[0], internal_energy(u0)};
93  auto [rho_c, E_c] = rhoEbar_cache(0);
94  auto delta_rhoE = RhoE{(rho - rho_c) / scale[0], (E - E_c) / scale[4]};
95 
96  if (zisa::norm(delta_rhoE) >= recompute_threshold) {
97  compute_equilibrium(u_local);
98  }
99  }
100  }

102  void compute(const array_view<double, 2, row_major> &rhs,
103  const array_view<WENOPoly, 1> &polys,
104  const array_view<cvars_t, 1> &u_local) {
105 
106  recompute_equilibrium(u_local);
107 
108  auto &l2g = rc.local2global();
109  for (int_t il = 0; il < l2g.size(); ++il) {
110  auto [rho_eq_bar, E_eq_bar] = equilibrium_cell_average(il);
111 
112  u_local(il)[0] -= rho_eq_bar;
113  u_local(il)[4] -= E_eq_bar;
114 
115  u_local(il) = u_local(il) / scale;
116  }
117 
118  weno_poly = rc.reconstruct(rhs, polys, u_local);
119  ++steps_since_recompute;
120  }

127  void compute_tracer(const array_view<double, 2, row_major> &rhs,
128  const array_view<ScalarPoly, 1> &polys,
129  const array_view<double, 2, column_major> &q_local) {
130 
131  auto n_vars = q_local.shape(1);
132  if (scalar_polys.size() != n_vars) {
133  scalar_polys = array<ScalarPoly, 1>(n_vars);
134  }
135 
136  auto rhs_view = [&]() {
137  auto shape = shape_t<2>{rhs.shape(0), 1};
138  return array_view<double, 2, row_major>(shape, rhs.raw());
139  }();
140 
141  for (int_t k_var = 0; k_var < n_vars; ++k_var) {
142  auto q_component = array_const_view<double, 1>(
143  shape_t<1>(q_local.shape(0)),
144  q_local.raw() + k_var * q_local.shape(0));
145  scalar_polys[k_var] = rc.reconstruct(rhs_view, polys, q_component);
146  }
