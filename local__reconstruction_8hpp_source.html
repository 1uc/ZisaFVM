23template <class Equilibrium, class RC, class Scaling>
24class LocalReconstruction {
25private:
26 using cvars_t = euler_var_t;
27 using xvars_t = euler_xvar_t;
28
29public:
30 LocalReconstruction() = default;
31 LocalReconstruction(std::shared_ptr<Grid> grid_,
32 const LocalEquilibrium<Equilibrium> &eq,
33 const RC &rc,
34 int_t i_cell,
35 Scaling scaling,
36 const LocalRCParams &params)
37 : grid(std::move(grid_)),
38 eq(eq),
39 rc(rc),
40 i_cell(i_cell),
41 scaling(scaling),
42 rhoEbar_cache(shape_t<1>{rc.local2global().size()}),
43 steps_per_recompute(params.steps_per_recompute),
44 recompute_threshold(params.recompute_threshold) {
45
46 auto n_points = grid->cells[i_cell].qr.points.size();
47 for (int_t k = 0; k < grid->max_neighbours; ++k) {
48 n_points += grid->face(i_cell, k).qr.points.size();
49 }
50
51 auto points = array<XYZ, 1>(n_points);
52 int_t k_point = 0;
53 for (const auto &p : grid->cells[i_cell].qr.points) {
54 points[k_point] = p;
55 ++k_point;
56 }
57
58 for (int_t k = 0; k < grid->max_neighbours; ++k) {
59 for (const auto &p : grid->face(i_cell, k).qr.points) {
60 points[k_point] = p;
61 ++k_point;
62 }
63 }
64
65 point_values_cache
66 = FewPointsCache<std::pair<RhoE, xvars_t>>(std::move(points));
67 }
68
69 void compute_equilibrium(const array_view<cvars_t, 1> &u_local) {
70 const auto &u0 = u_local(int_t(0));
71 auto rhoE_self = RhoE{u0[0], internal_energy(u0)};
72
73 auto &l2g = rc.local2global();
74 scale = scaling(rhoE_self);
75 eq.solve(rhoE_self, grid->cells(i_cell));
76
77 point_values_cache.update(
78 [this](const XYZ &x) { return eq.extrapolate_full(x); });
79
80 for (int_t il = 0; il < l2g.size(); ++il) {
81 rhoEbar_cache(il) = eq.extrapolate(grid->cells(l2g[il]));
82 }
83
84 steps_since_recompute = 0;
85 }
86
87 void recompute_equilibrium(const array_view<cvars_t, 1> &u_local) {
88 if (steps_since_recompute % steps_per_recompute == 0) {
89 compute_equilibrium(u_local);
90 } else {
91 const auto &u0 = u_local(int_t(0));
92 auto [rho, E] = RhoE{u0[0], internal_energy(u0)};
93 auto [rho_c, E_c] = rhoEbar_cache(0);
94 auto delta_rhoE = RhoE{(rho - rho_c) / scale[0], (E - E_c) / scale[4]};
95
96 if (zisa::norm(delta_rhoE) >= recompute_threshold) {
97 compute_equilibrium(u_local);
98 }
99 }
100 }
101
102 void compute(const array_view<double, 2, row_major> &rhs,
103 const array_view<WENOPoly, 1> &polys,
104 const array_view<cvars_t, 1> &u_local) {
105
106 recompute_equilibrium(u_local);
107
108 auto &l2g = rc.local2global();
109 for (int_t il = 0; il < l2g.size(); ++il) {
110 auto [rho_eq_bar, E_eq_bar] = equilibrium_cell_average(il);
111
112 u_local(il)[0] -= rho_eq_bar;
113 u_local(il)[4] -= E_eq_bar;
114
115 u_local(il) = u_local(il) / scale;
116 }
117
118 weno_poly = rc.reconstruct(rhs, polys, u_local);
119 ++steps_since_recompute;
120 }
121
122 RhoE equilibrium_cell_average(int_t il) { return rhoEbar_cache(il); }
123 const auto &equilibrium_points_values(const XYZ &x) {
124 return point_values_cache.get(x);
125 }

127 void compute_tracer(const array_view<double, 2, row_major> &rhs,
128 const array_view<ScalarPoly, 1> &polys,
129 const array_view<double, 2, column_major> &q_local) {
130
131 auto n_vars = q_local.shape(1);
132 if (scalar_polys.size() != n_vars) {
133 scalar_polys = array<ScalarPoly, 1>(n_vars);
134 }
135
136 auto rhs_view = [&]() {
137 auto shape = shape_t<2>{rhs.shape(0), 1};
138 return array_view<double, 2, row_major>(shape, rhs.raw());
139 }();
140
141 for (int_t k_var = 0; k_var < n_vars; ++k_var) {
142 auto q_component = array_const_view<double, 1>(
143 shape_t<1>(q_local.shape(0)),
144 q_local.raw() + k_var * q_local.shape(0));
145 scalar_polys[k_var] = rc.reconstruct(rhs_view, polys, q_component);
146 }
147 }
148
149 cvars_t operator()(const XYZ &x) const {
150 return cvars_t(background(x).first + delta(x));
151 }
152
153 double tracer(const XYZ &x, int_t k_var) const {
154 return scalar_polys[k_var](x)[0];
155 }
156
157 cvars_t delta(const XYZ &x) const { return cvars_t(scale * weno_poly(x)); }
158
159 std::pair<cvars_t, xvars_t> background(const XYZ &x) const {
160 auto [rhoE, w] = point_values_cache.get(x);
161 auto u = cvars_t{rhoE.rho(), 0.0, 0.0, 0.0, rhoE.E()};
162 return {u, w};
163 }
164
165 auto combined_stencil_size() const
166 -> decltype(std::declval<RC>().combined_stencil_size()) {
167 return rc.combined_stencil_size();
168 }
169
170 const std::vector<int_t> &local2global() const { return rc.local2global(); }

172 std::string str(int verbose = 0) const {
173 std::stringstream ss;
174
175 ss << grid->str() << "\n";
176 ss << eq.str(verbose) << "\n";
177 ss << rc.str(verbose) << "\n";
178
179 return ss.str();
180 }
181
182private:
183 std::shared_ptr<Grid> grid;
184 LocalEquilibrium<Equilibrium> eq;
185 RC rc;
186 int_t i_cell;
187 WENOPoly weno_poly;
188 array<ScalarPoly, 1> scalar_polys;
189 Scaling scaling;
190 cvars_t scale = cvars_t::zeros();
191
192 array<RhoE, 1> rhoEbar_cache;
193 FewPointsCache<std::pair<RhoE, xvars_t>> point_values_cache;
194 int_t steps_per_recompute;
195 int_t steps_since_recompute = 0;
196 double recompute_threshold;
197};
