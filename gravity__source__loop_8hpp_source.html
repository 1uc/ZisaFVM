13template <class Equilibrium, class RC, class LEOS, class Gravity, class Scaling>
14class GravitySourceLoop : public RateOfChange {
15private:
16 using euler_t = Euler;
17 using leos_t = LEOS;
18 using gravity_t = Gravity;
19 using cvars_t = euler_var_t;
20 using grc_t = EulerGlobalReconstruction<Equilibrium, RC, Scaling>;
21
22public:
23 GravitySourceLoop(std::shared_ptr<Grid> grid,
24 std::shared_ptr<leos_t> local_eos,
25 std::shared_ptr<gravity_t> gravity,
26 std::shared_ptr<grc_t> global_reconstruction)
27 : local_eos(std::move(local_eos)),
28 gravity(std::move(gravity)),
29 grid(std::move(grid)),
30 global_reconstruction(std::move(global_reconstruction)) {}
31
32 virtual void compute(AllVariables &tendency,
33 const AllVariables & /* current_state */,
34 double /* t */) const override {
35
36 auto f = [this, &tendency](int_t i, const Cell &cell) {
37 const auto &rc = (*global_reconstruction)(i);
38 const auto &eos = *(*local_eos)(i);
39
40 auto x_cell = grid->cell_centers(i);
41
42 // equilibrium terms
43 auto s = cvars_t(0.0);
44 for (int_t k = 0; k < grid->max_neighbours; ++k) {
45 auto face = grid->face(i, k);
46
47 auto s_eq = [&x_cell, &rc, &eos, &face](const XYZ &x) {
48 auto [u_eq, w_eq] = rc.background(x);
49 auto p_eq = w_eq.p;
50
51 auto n = unit_outward_normal(face, x_cell);
52 auto s = cvars_t{0.0, p_eq * n[0], p_eq * n[1], p_eq * n[2], 0.0};
53
54 return s;
55 };
56
57 s += quadrature(face, s_eq);
58 }
59
60 // delta terms
61 auto s_delta = [&rc, &gravity = *this->gravity](const XYZ &x) {
62 static_assert(XYZ::size() == 3);
63
64 auto [u_eq, _] = rc.background(x);
65 auto du = rc.delta(x);
66 auto u = cvars_t(u_eq + du);
67
68 auto drho = du[0];
69 auto mv = momentum(u);
70 auto grad_phi = gravity.grad_phi(x);
71 static_assert(decltype(grad_phi)::size() == 3);
72
73 auto s = cvars_t{0.0,
74 -drho * grad_phi[0],
75 -drho * grad_phi[1],
76 -drho * grad_phi[2],
77 -zisa::dot(mv, grad_phi)};
78
79 return s;
80 };
81
82 s += quadrature(cell, s_delta);
83 tendency.cvars(i) += s / volume(cell);
84 };
85
86 zisa::for_each(cells(*grid), f);
87 }
88
89 virtual std::string str() const override {
90 return "Gravity source term:\n" + indent_block(1, type_name<Equilibrium>())
91 + "\n" + indent_block(1, type_name<RC>());
92 }
93
94private:
95 std::shared_ptr<leos_t> local_eos;
96 std::shared_ptr<gravity_t> gravity;
97 std::shared_ptr<Grid> grid;
98 std::shared_ptr<grc_t> global_reconstruction;
99};

101template <class RC, class EOS, class Gravity, class Scaling>
102class GravitySourceLoop<NoEquilibrium, RC, EOS, Gravity, Scaling>
103 : public RateOfChange {
104
105private:
106 using euler_t = Euler;
107 using eos_t = EOS;
108 using gravity_t = Gravity;
109 using cvars_t = euler_var_t;
110 using grc_t = EulerGlobalReconstruction<NoEquilibrium, RC, Scaling>;
111
112public:
113 GravitySourceLoop(std::shared_ptr<Grid> grid,
114 std::shared_ptr<eos_t>,
115 std::shared_ptr<gravity_t> gravity,
116 std::shared_ptr<grc_t> global_reconstruction)
117 : gravity(std::move(gravity)),
118 grid(std::move(grid)),
119 global_reconstruction(std::move(global_reconstruction)) {}
120
121 virtual void compute(AllVariables &tendency,
122 const AllVariables & /* current_state */,
123 double /* t */) const override {
124
125 auto f = [this, &tendency](int_t i, const Cell &cell) {
126 const auto &rc = (*global_reconstruction)(i);
127
128 auto s = [&rc, &gravity = *this->gravity](const XYZ &x) {
129 static_assert(XYZ::size() == 3);
130
131 auto u = rc(x);
132 auto rho = u[0];
133 auto mv = momentum(u);
134 auto grad_phi = gravity.grad_phi(x);
135 static_assert(decltype(grad_phi)::size() == 3);
136
137 return cvars_t{0.0,
138 -rho * grad_phi[0],
139 -rho * grad_phi[1],
140 -rho * grad_phi[2],
141 -zisa::dot(mv, grad_phi)};
142 };
143
144 tendency.cvars(i) += average(cell, s);
145 };
146
147 zisa::for_each(cells(*grid), f);
148 }
149
150 virtual std::string str() const override {
151 return "Gravity source term:\n"
152 + indent_block(1, type_name<NoEquilibrium>()) + "\n"
153 + indent_block(1, type_name<RC>());
154 }
155
156private:
157 std::shared_ptr<gravity_t> gravity;
158 std::shared_ptr<Grid> grid;
159 std::shared_ptr<grc_t> global_reconstruction;
160};
