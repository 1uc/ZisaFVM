34 public:
35  JankaEOS() = default;
36  JankaEOS(double rho_bounce,
37  const std::array<double, 2> &gamma,
38  double gamma_thermal,
39  double E1)
40  : rho_bounce(rho_bounce), gamma(gamma), gamma_thermal(gamma_thermal) {
41 
42  E[0] = E1;
43  E[1] = (gamma[0] - 1.0) / (gamma[1] - 1.0) * E1
44  * pow(rho_bounce, gamma[0] - gamma[1]);
45 
46  E[2] = (gamma[1] - gamma[0]) / (gamma[1] - 1.0) * E1
47  * pow(rho_bounce, gamma[0] - 1.0);
48 
49  K[0] = (gamma[0] - 1.0) * E[0];
50  K[1] = (gamma[1] - 1.0) * E[1];
51 
52  h_rho_bounce = enthalpy_fixed<0>(rho_bounce);
53  }
54

66  // --- Ideal Gas EOS calls ---------------------------------------------------
67  double ideal_gas_enthalpy(double K, double rho, double gamma) const {
68  return gamma / (gamma - 1.0) * K * pow(rho, gamma - 1.0);
69  }
70 
72  double ideal_gas_pressure(double K, double rho, double gamma) const {
73  return K * zisa::pow(rho, gamma);
74  }
75 
76  // --- gamma ----------------------------------------------------------------
77  double polytropic_gamma(double rho) const {
78  return rho <= rho_bounce ? gamma[0] : gamma[1];
79  }
80 
81  // --- Density ---------------------------------------------------------------
82  double ideal_gas_density(double gamma, const EnthalpyEntropy &theta) const {
83  const auto &[h, K] = theta;
84  return zisa::pow((gamma - 1) / gamma * h / K, 1.0 / (gamma - 1));
85  }
86 
87  template <int REGIME>
88  double rho_fixed(EnthalpyEntropy theta) const {
89  const auto &[h, K] = theta;
90 
91  double K_p = this->K[REGIME];
92  double K_th = zisa::max(0.0, K - K_p);
93  // double K_th = K - K_p;
94 
95  auto f = [this, h = h, K_th](double rho) {
96  double h_p = enthalpy_fixed<REGIME>(rho);
97  double h_th = ideal_gas_enthalpy(K_th, rho, gamma_thermal);
98 
99  return h - (h_p + h_th);
100  };
101 
102  double inf = std::numeric_limits<double>::infinity();
103  auto hard_bounds = (REGIME == 0 ? std::array<double, 2>{0.0, rho_bounce}
104  : std::array<double, 2>{rho_bounce, inf});
105 
106  double rho_a = ideal_gas_density(gamma[REGIME], theta);
107  rho_a = zisa::max(hard_bounds[0], rho_a);
108  double rho_b = zisa::min(hard_bounds[1], 2.0 * rho_a);
109 
110  double atol = 1e-10 * zisa::avg(rho_a, rho_b);
111  int max_iter = 100;
112 
113  std::tie(rho_a, rho_b)
114  = find_bracket(f, {rho_a, rho_b}, hard_bounds, max_iter);
115 
116  LOG_ERR_IF(f(rho_a) * f(rho_b) < 0.0, "Failed to compute a valid bracket.");
117 
118  return brent(f, rho_a, rho_b, atol, max_iter);
119  }

121  ANY_DEVICE_INLINE double rho(EnthalpyEntropy theta) const {
122  const auto &[h, K] = theta;
123 
124  if (h <= h_rho_bounce) {
125  // definitely rho < rho_b
126  return this->rho_fixed<0>(theta);
127  } else {
128  // definitely rho > rho_b
129  return this->rho_fixed<1>(theta);
130  }
131  }

133  // --- Pressure -------------------------------------------------------------
134  ANY_DEVICE_INLINE double polytropic_pressure(double rho) const {
135  int_t i = rho <= rho_bounce ? 0 : 1;
136  return K[i] * zisa::pow(rho, gamma[i]);
137  }

139  ANY_DEVICE_INLINE double thermal_pressure(double thermal_energy) const {

143  ANY_DEVICE_INLINE double thermal_pressure(const RhoP &rhoP) const {
144  const auto &[rho, p] = rhoP;
145  return zisa::max(0.0, p - polytropic_pressure(rho));
146  // return p - polytropic_pressure(rho);

149  ANY_DEVICE_INLINE double total_pressure(double rho, double E_th) const {

159  ANY_DEVICE_INLINE double pressure(RhoEntropy rhoK) const {
160  const auto &[rho, K] = rhoK;
161 
162  // double K_thermal = K - polytropic_entropy(rho);
163  double K_thermal = zisa::max(0.0, K - polytropic_entropy(rho));
164  double p_thermal = ideal_gas_pressure(K_thermal, rho, gamma_thermal);
165 
166  return polytropic_pressure(rho) + p_thermal;

178  ANY_DEVICE_INLINE std::pair<double, double>

188  return p_thermal / (gamma_thermal - 1.0);

200  ANY_DEVICE_INLINE double polytropic_energy(double rho) const {
201  if (rho <= rho_bounce) {
202  return E[0] * pow(rho, gamma[0]);
203  } else {
204  return E[1] * pow(rho, gamma[1]) + E[2] * rho;
205  }
206  }

209  const auto &[rho, p] = rhoP;

222  // --- Enthalpy --------------------------------------------------------------
223  template <int REGIME>
224  double enthalpy_fixed(double rho) const {
225  double h_p = ideal_gas_enthalpy(K[REGIME], rho, gamma[REGIME]);
226  return h_p + (REGIME == 0 ? 0.0 : E[2]);
227  }

229  double polytropic_enthalpy(double rho) const {
230  if (rho <= rho_bounce) {
231  return enthalpy_fixed<0>(rho);
232  } else {
233  return enthalpy_fixed<1>(rho);
234  }
235  }

237  ANY_DEVICE_INLINE double enthalpy(const RhoP &rhoP) const {
238  const auto &[rho, p] = rhoP;
239  double E = internal_energy(rhoP);
240  return (E + p) / rho;

256  ANY_DEVICE_INLINE double entropy(const RhoP &rhoP) const {
257  const auto &[rho, p] = rhoP;
258  const auto &[p_polytropic, p_thermal] = split_pressure(rhoP);
259 
260  double K_polytropic = polytropic_entropy(rho);
261  double K_thermal = p_thermal * zisa::pow(rho, -gamma_thermal);
262 
263  return K_polytropic + K_thermal;
264  }

266  // --- speed of sound --------------------------------------------------------
267  ANY_DEVICE_INLINE double sound_speed(const RhoE &rhoE) const {

271  ANY_DEVICE_INLINE double sound_speed(const RhoP &rhoP) const {
272  double rho = rhoP.rho();
273  double K = entropy(rhoP);
274  double K_p = polytropic_entropy(rho);
275  // double K_th = K - K_p;
276  double K_th = zisa::max(0.0, K - K_p);
277 
278  double gamma_p = polytropic_gamma(rho);
279 
280  double cs2 = gamma_p * K_p * zisa::pow(rho, gamma_p - 1.0)
281  + gamma_thermal * K_th * zisa::pow(rho, gamma_thermal - 1.0);
282 
283  return zisa::sqrt(cs2);
284  }
