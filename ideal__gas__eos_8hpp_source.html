162 ANY_DEVICE_INLINE double rho(EnthalpyEntropy theta) const {
163 const auto &[h, K] = theta;
164 double gamma = IdealGasEOS::gamma();
165 double base = 1.0 / K * (gamma - 1.0) / gamma * h;
166 double exponent = 1.0 / (gamma - 1.0);
167 return zisa::pow(base, exponent);
168 }
169
170 ANY_DEVICE_INLINE double rho(PressureEntropy theta) const {
171 const auto &[p, K] = theta;
172 return zisa::pow(p / K, 1.0 / gamma());
173 }

177 ANY_DEVICE_INLINE double pressure(EnthalpyEntropy theta) const {

184 ANY_DEVICE_INLINE double pressure(RhoE rhoE) const {

188 ANY_DEVICE_INLINE double pressure(RhoEntropy rhoK) const {

193 ANY_DEVICE_INLINE double pressure(RhoT rhoT) const {

196 return r_gas * rho * T;

258 euler_full_xvars_t full_extra_variables(const RhoE &rhoE) const {
259 euler_full_xvars_t ret;
260
261 auto [rho, E] = rhoE;
262 auto p = pressure(rhoE);
263 auto a = sound_speed(RhoP{rho, p});
264 auto h = enthalpy(RhoP{rho, p});
265 auto s = entropy(RhoP{rho, p});
266 auto T = temperature(RhoP{rho, p});
267
268 ret.rho = rho;
269 ret.E = E;
270 ret.a = a;
271 ret.h = h;
272 ret.s = s;
273 ret.p = p;
274 ret.T = T;
275
276 return ret;
277 }
