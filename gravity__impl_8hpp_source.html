30// --- Polytrope ---------------------------------------------------
31ANY_DEVICE_INLINE double PolytropeGravity::phi(double chi) const {
32 double alpha = PolytropeGravity::alpha(chi);
33
34 // Potential division by zero coming up:
35 // lim_{r->0} sin(r)/r = 1
36 // The problem only exists for `r == 0`. The correct limiting behaviour
37 // is guaranteed. we therefore add a small `eps` which will almost always be
38 // truncated, except for the case of `r == 0`.
39 double chi_eff = alpha * (chi + eps);
40
41 return -2.0 * K * rhoC * zisa::sin(chi_eff) / chi_eff;
42}
43
44ANY_DEVICE_INLINE double PolytropeGravity::dphi_dx(double chi) const {
45 double alpha = PolytropeGravity::alpha(chi);
46 double chi_eff = alpha * (chi + eps);
47
48 double dphi = (zisa::cos(chi_eff) - zisa::sin(chi_eff) / chi_eff) / chi_eff;
49 return -2.0 * K * rhoC * dphi * alpha;
50}

70inline PolytropeGravityWithJump::PolytropeGravityWithJump(
71 double r_crit, double rhoC, double K_inner, double K_outer, double G)
72 : r_crit(r_crit), inner(rhoC, K_inner, G), outer(1.0, K_outer, G) {
73
74 double rhoC_outer = inner.phi(r_crit) / outer.phi(r_crit);
