40class LocalEOSState<HelmholtzEOS> {

44public:
45 LocalEOSState() = delete;
46 LocalEOSState(int_t n_cells,
47 std::vector<double> mass_number,
48 std::vector<double> charge_number)
49 : local_eos(shape_t<1>(n_cells)),
50 mass_fraction(mass_number.size()),
51 mass_number(std::move(mass_number)),
52 charge_number(std::move(charge_number)) {
53
54 for (int_t i = 0; i < n_cells; ++i) {
55 local_eos[i] = std::make_shared<HelmholtzEOS>();
56 }
57 }
58
59 void compute(const AllVariables &all_variables) {
60 const auto &avars = all_variables.avars;
61 const auto &cvars = all_variables.cvars;
62
63 auto n_cells = avars.shape(0);
64 auto n_avars = avars.shape(1);
65
66 LOG_ERR_IF(n_cells != local_eos.shape(0),
67 string_format(
68 "Sizes don't match: %d != %d", n_cells, local_eos.shape(0)));
69
70 for (int_t i = 0; i < n_cells; ++i) {
71 for (int_t k = 0; k < n_avars; ++k) {
72 mass_fraction[k] = zisa::max(0.0, avars(i, k) / cvars(i, 0));
73 }
74
75 *local_eos[i] = HelmholtzEOS(mass_fraction, mass_number, charge_number);
76 }
77 }
78
79 void compute(const AllVariables &all_variables,
80 const array_const_view<int_t, 1> &cells) {
81
82 const auto &avars = all_variables.avars;
83 const auto &cvars = all_variables.cvars;
84
85 auto n_cells = cells.size();
86 auto n_avars = avars.shape(1);
87
88 LOG_ERR_IF(avars.shape(0) != local_eos.shape(0),
89 string_format(
90 "Sizes don't match: %d != %d", n_cells, local_eos.shape(0)));
91
92 // TODO not OpenMP parallel
93 for (int_t ii = 0; ii < n_cells; ++ii) {
94 auto i = cells[ii];
95 for (int_t k = 0; k < n_avars; ++k) {
96 mass_fraction[k] = zisa::max(0.0, avars(i, k) / cvars(i, 0));
97 }
98
99 *local_eos[i] = HelmholtzEOS(mass_fraction, mass_number, charge_number);
100 }
101 }
102
103 std::shared_ptr<HelmholtzEOS> operator()(int_t i) const {
104 return local_eos[i];
105 }
106
107private:
108 array<std::shared_ptr<HelmholtzEOS>, 1> local_eos;
109 std::vector<double> mass_fraction;
110 std::vector<double> mass_number;
111 std::vector<double> charge_number;
