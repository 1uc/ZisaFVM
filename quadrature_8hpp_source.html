33 template <class QR, class F, class Domain>
34 auto quadrature(const QR &qr, const F &f, const Domain &domain)
35  -> decltype(f(std::declval<XYZ>())) {
36 
37  using fx_t = decltype(f(std::declval<XYZ>()));
38 
39  const auto &w = qr.weights;
40  const auto &x = qr.points;
41 
42  // avoids assigning 'zero' to `ret`.
43  auto ret = fx_t(w[0] * f(coord(domain, x[0])));
44 
45  // starts at --v
46  for (int_t i = 1; i < qr.weights.size(); ++i) {
47  ret = fx_t(ret + w[i] * f(coord(domain, x[i])));
48  }
49 
50  return fx_t(volume(domain) * ret);
51 }
