33template <class QR, class F, class Domain>
34auto quadrature(const QR &qr, const F &f, const Domain &domain)
35 -> decltype(f(std::declval<XYZ>())) {
36
37 using fx_t = decltype(f(std::declval<XYZ>()));
38
39 const auto &w = qr.weights;
40 const auto &x = qr.points;
41
42 // avoids assigning 'zero' to `ret`.
43 auto ret = fx_t(w[0] * f(coord(domain, x[0])));
44
45 // starts at --v
46 for (int_t i = 1; i < qr.weights.size(); ++i) {
47 ret = fx_t(ret + w[i] * f(coord(domain, x[i])));
48 }
49
50 return fx_t(volume(domain) * ret);
51}

115template <class F>
116auto quadrature(const F &f, const Grid &grid)
117 -> decltype(f(std::declval<XYZ>())) {
118
119 // avoids assigning 'zero' to `ret`.
120 auto ret = quadrature(grid.cells(0), f);
121
122 // starts at --v
123 for (int_t i : PlainIndexRange(1, grid.n_cells)) {
124 ret = ret + quadrature(grid.cells(i), f);
125 }
126
127 return ret;
128}
