34  void compute(const array_view<double, 2, row_major> &rhs,
35  const array_view<WENOPoly, 1> &polys,
36  const array_view<cvars_t, 1> &u_local) {
37 
38  if (steps_since_recompute % steps_per_recompute == 0) {
39  recompute_cache(rhs, polys, u_local);
40  } else {
41  const auto &u0 = u_local(int_t(0));
42  auto [rho, E] = RhoE{u0[0], internal_energy(u0)};
43  auto [rho_c, E_c] = rhoE_eq_cache(0);
44  auto elta_rhoE = RhoE{(rho - rho_c) / scale[0],
45  (E - E_c) / scale[4]};
46 
47  if (zisa::norm(delta_rhoE) >= recompute_threshold) {
48  recompute_cache(rhs, polys, u_local);
49  }
50  }
51 
52  auto &l2g = rc.local2global();
53  for (int_t il = 0; il < l2g.size(); ++il) {
54  auto [rho_eq_bar, E_eq_bar] = rhoE_eq_cache(il);
55 
56  u_local(il)[0] -= rho_eq_bar;
57  u_local(il)[4] -= E_eq_bar;
58 
59  u_local(il) = u_local(il) / scale;
60  }
61 
62  weno_poly = rc.reconstruct(rhs, polys, u_local);
63 
64  ++steps_since_recomputes;
65  }

68  void recompute_cache(const array_view<double, 2, row_major> &rhs,
69  const array_view<WENOPoly, 1> &polys,
70  const array_view<cvars_t, 1> &u_local) {
71 
72  const auto &u0 = u_local(int_t(0));
73  auto rhoE_self = RhoE{u0[0], internal_energy(u0)};
74 
75  scale = scaling(rhoE_self);
76  eq.solve(rhoE_self, grid->cells(i_cell));
77  auto &l2g = rc.local2global();
78  for (int_t il = 0; il < l2g.size(); ++il) {
79  rhoE_eq_cache(il) = eq.extrapolate(grid->cells(l2g[il]));
80  }
81 
82  steps_since_recompute = 0;
83  }

85  void compute_tracer(const array_view<double, 2, row_major> &rhs,
86  const array_view<ScalarPoly, 1> &polys,
87  const array_view<double, 2, column_major> &q_local) {
88

90  if (scalar_polys.size() != n_vars) {
91  scalar_polys = array<ScalarPoly, 1>(n_vars);
92  }
93 
94  auto rhs_view = [&]() {
95  auto shape = shape_t<2>{rhs.shape(0), 1};
96  return array_view<double, 2, row_major>(shape, rhs.raw());
97  }();
98 
99  for (int_t k_var = 0; k_var < n_vars; ++k_var) {
100  auto q_component = array_const_view<double, 1>(
101  shape_t<1>(q_local.shape(0)),
102  q_local.raw() + k_var * q_local.shape(0));
103  scalar_polys[k_var] = rc.reconstruct(rhs_view, polys, q_component);
104  }
105  }
