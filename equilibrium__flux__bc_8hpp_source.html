36 virtual void compute(AllVariables &tendency,
37 const AllVariables &current_state,
38 double /* t */) const override {
39
40 const auto &euler = *this->euler;
41
42 for (auto &&[e, face] : exterior_faces(*grid)) {
43 auto i = grid->left_right(e).first;
44 const auto &cell = grid->cells(i);
45 const auto &eos = (*local_eos)(i);
46
47 auto eq = LocalEquilibrium(Equilibrium(eos, gravity));
48 eq.solve(eos->rhoE(cvars_t(current_state.cvars(i))), cell);
49
50 auto flux = [&euler, &eos = *eos, &eq, &face = face](XYZ x) {
51 auto u = eos.cvars(eq.extrapolate(x));
52 auto xvars = eos.xvars(u);
53 auto f = euler.flux(u, xvars.p);
54 inv_coord_transform(f, face);
55
56 return f;
57 };
58
59 auto f = quadrature(face, flux);
60 tendency.cvars(i) -= f / volume(cell);
61 }
62 }
