17 template <class GravityBase, class Alignment>
18 class Gravity {
19 public:
20  Gravity() = default;
21  Gravity(GravityBase gravity, Alignment alignment)
22  : gravity(std::move(gravity)), alignment(std::move(alignment)) {}
23 
24  ANY_DEVICE_INLINE double phi(const XYZ &x) const {
25  double chi = alignment.coordinate(x);
26  return gravity.phi(chi);
27  }
28 
29  ANY_DEVICE_INLINE double dphi_dx(const XYZ &x, int_t dir) const {
30  double chi = alignment.coordinate(x);
31  return gravity.dphi_dx(chi) * alignment.dx(x, dir);
32  }
33 
34  ANY_DEVICE_INLINE XYZ grad_phi(const XYZ &x) const {
35  return {dphi_dx(x, 0), dphi_dx(x, 1), dphi_dx(x, 2)};
36  }
37 
38  ANY_DEVICE_INLINE double norm_grad_phi(const XYZ &x) const {
39  return zisa::norm(grad_phi(x));
40  }
41 
42  std::string str() const {
43  return string_format("%s with %s",
44  type_name<GravityBase>().c_str(),
45  type_name<Alignment>().c_str());
46  }

51  template <class RetGravity>
52  [[nodiscard]] static RetGravity load_impl(HierarchicalReader &reader) {
53  reader.open_group("gravity");
54  auto g = GravityBase::load(reader);
55  reader.switch_group("alignment");
56  auto a = Alignment::load(reader);
57  reader.close_group();
58 
59  return RetGravity(
60  Gravity<GravityBase, Alignment>(std::move(g), std::move(a)));
61  }
62 
63 protected:
64  GravityBase gravity;
65  Alignment alignment;
66 };

70  writer.open_group("gravity");
71  save(writer, gravity.gravity);
72  writer.switch_group("alignment");
73  save(writer, gravity.alignment);
74  writer.close_group();
