14template <class EOS, class Gravity>
15class PolytropeIC {
16private:
17 using eos_t = EOS;
18 using gravity_t = Gravity;
19
20public:
21 PolytropeIC(std::shared_ptr<eos_t> eos, std::shared_ptr<gravity_t> gravity)
22 : eos(std::move(eos)), gravity(std::move(gravity)) {}
23
24 RhoP operator()(const XYZ &x) const {
25
26 double alpha = gravity->alpha(zisa::norm(x));
27 double eps = std::numeric_limits<double>::min();
28 double r_eff = alpha * (zisa::norm(x) + eps);
29
30 double rho = zisa::sin(r_eff) / r_eff;
31 double p = zisa::pow<2>(rho);
32
33 return RhoP{rho, p};
34 }
35
36private:
37 std::shared_ptr<eos_t> eos;
38 std::shared_ptr<gravity_t> gravity;
39};

67template <class EOS, class Gravity>
68class PolytropeWithJumpIC {
69private:
70 using eos_t = EOS;
71 using gravity_t = Gravity;
72 using eq_t = IsentropicEquilibrium<eos_t, gravity_t>;
73
74public:
75 PolytropeWithJumpIC(const std::shared_ptr<eos_t> &eos,
76 const std::shared_ptr<gravity_t> &gravity,
77 const EnthalpyEntropy &theta_inner,
78 const EnthalpyEntropy &theta_outer,
79 const XYZ &x_ref,
80 double amp_shape,
81 int n_bumps)
82 : eos(eos),
83 x_ref(x_ref),
84 inner_equilibrium(eq_t(eos, gravity), theta_inner, x_ref),
85 outer_equilibrium(eq_t(eos, gravity), theta_outer, x_ref),
86 amp_shape(amp_shape),
87 n_bumps(n_bumps) {}
88
89 RhoP operator()(const XYZ &x) const {
90 const auto &eq = (is_inner(x) ? inner_equilibrium : outer_equilibrium);
91 RhoE rhoE = eq.extrapolate(x);
92 return eos->rhoP(rhoE);
93 }
94
95 bool is_inner(const XYZ &x) const {
96 double alpha = n_bumps * zisa::angle(x);
97 double r_ref = zisa::norm(x_ref) + amp_shape * zisa::sin(alpha);
98
99 return zisa::norm(x) < r_ref;
100 }
101
102private:
103 std::shared_ptr<eos_t> eos;
104 XYZ x_ref;
105 LocalEquilibrium<eq_t> inner_equilibrium;
106 LocalEquilibrium<eq_t> outer_equilibrium;
107
108 double amp_shape;
109 int n_bumps;
110};
