13template <class Range, class Body>
14void for_each(omp_policy, const Range &range, const Body &body) {
15 auto i0 = range.start_index();
16 auto i_end = range.end_index();
17
18#if ZISA_HAS_OPENMP == 1
19#pragma omp parallel for ZISA_OMP_FOR_SCHEDULE_DEFAULT
20#endif
21 for (auto i = i0; i < i_end; ++i) {
22 if constexpr (range_traits<Range>::has_item) {
23 body(i, range.item(i));
24 } else {
25 body(i);
26 }
27 }
28}
