40template <class Equilibrium>
41void LocalEquilibriumBase<Equilibrium>::solve_exact(const RhoE &rhoE_bar,
42 const Cell &cell_ref) {
43
44 x_ref = cell_ref.qr.points[0];
45
46 const auto &eos = *equilibrium.eos;
47
48 auto full_guess = eos.full_extra_variables(rhoE_bar);
49 auto enthalpy_entropy_guess = EnthalpyEntropy{full_guess.h, full_guess.s};
50 auto rhoT_guess = RhoT{full_guess.rho, full_guess.T};
51
52 auto f = [this, &eos, &cell_ref, &rhoE_bar, &rhoT_guess](
53 const EnthalpyEntropy &theta_star) {
54 auto rhoE_eq = [this, &eos, &theta_star, &rhoT_guess](const XYZ &xy) {
55 return equilibrium.extrapolate(
56 isentropic_equilibrium_values(eos, theta_star, rhoT_guess),
57 x_ref,
58 xy);
59 };
60
61 return RhoE(rhoE_bar - average(cell_ref, rhoE_eq));
62 };
63
64 auto df = [&f](const EnthalpyEntropy &x, int_t dir) {
65 double eps = 1e-6 * zisa::abs(x[dir]);
66
67 auto x_p_eps
68 = static_cast<EnthalpyEntropy>(x + 0.5 * eps * x.unit_vector(dir));
69
70 auto x_m_eps
71 = static_cast<EnthalpyEntropy>(x - 0.5 * eps * x.unit_vector(dir));
72
73 return RhoE((f(x_p_eps) - f(x_m_eps)) / eps);
74 };
75
76 auto inv_df = [&df](const EnthalpyEntropy &x) {
77 auto df0 = df(x, 0ul);
78 auto df1 = df(x, 1ul);
79
80 return [df0, df1](const auto &fx) {
81 double inv_det = 1.0 / (df0[0] * df1[1] - df0[1] * df1[0]);
82
83 return EnthalpyEntropy{inv_det * (df1[1] * fx[0] - df1[0] * fx[1]),
84 inv_det * (-df0[1] * fx[0] + df0[0] * fx[1])};
85 };
86 };
87
88 auto atol = EnthalpyEntropy(1e-13 * enthalpy_entropy_guess);
89
90 auto [hS, has_eq] = quasi_newton(f, inv_df, enthalpy_entropy_guess, atol);
91
92 found_equilibrium = has_eq;
93 theta = isentropic_equilibrium_values(eos, hS, rhoT_guess);
94}

98class LMDRhoEBar : public Eigen::DenseFunctor<double> {
99public:
100 LMDRhoEBar(const std::function<RhoE(const EnthalpyEntropy &)> &f_)
101 : DenseFunctor<double>(2, 2), f(f_) {}
102
103 int operator()(const Eigen::VectorXd &theta,
104 Eigen::VectorXd &drhoE_bar) const {
105
106 auto drhoE_bar_ = f(EnthalpyEntropy{theta[0], theta[1]});
107
108 drhoE_bar[0] = drhoE_bar_.rho();
109 drhoE_bar[1] = drhoE_bar_.E();
110
111 return 0;
112 }
113
114private:
115 std::function<RhoE(const EnthalpyEntropy &)> f;
116};

120template <class Equilibrium>
121void LocalEquilibriumBase<Equilibrium>::solve_lsq(const RhoE &rhoE_bar,
122 const Cell &cell_ref) {
123
124 x_ref = cell_ref.qr.points[0];
125
126 const auto &eos = *equilibrium.eos;
127
128 auto full_guess = eos.full_extra_variables(rhoE_bar);
129 auto rhoT_guess = RhoT{full_guess.rho, full_guess.T};
130
131 auto drhoE_bar = [this, &eos, &cell_ref, &rhoE_bar, &rhoT_guess](
132 const EnthalpyEntropy &theta_star) {
133 auto rhoE_eq = [this, &eos, &theta_star, &rhoT_guess](const XYZ &xy) {
134 return equilibrium.extrapolate(
135 isentropic_equilibrium_values(eos, theta_star, rhoT_guess),
136 x_ref,
137 xy);
138 };
139
140 return RhoE(rhoE_bar - average(cell_ref, rhoE_eq));
141 };
142
143 auto lm_drhoE_bar = detail::LMDRhoEBar(drhoE_bar);
144 auto lm_drhoE_bar_der
145 = Eigen::NumericalDiff<detail::LMDRhoEBar>(lm_drhoE_bar);
146 auto lm = Eigen::LevenbergMarquardt<Eigen::NumericalDiff<detail::LMDRhoEBar>>(
147 lm_drhoE_bar_der);
148
149 auto hS = Eigen::VectorXd(2);
150 hS[0] = full_guess.h;
151 hS[1] = full_guess.s;
152 auto info = lm.minimize(hS);
153 LOG_ERR_IF(info != 1, "`LevenbergMarquardt` failed.");
154
155 found_equilibrium = true;
156 theta = isentropic_equilibrium_values(
157 eos, EnthalpyEntropy{hS[0], hS[1]}, rhoT_guess);
158}

160template <class Equilibrium>
161RhoE LocalEquilibriumBase<Equilibrium>::extrapolate(const XYZ &xy) const {
162 if (found_equilibrium) {
163 return equilibrium.extrapolate(theta, x_ref, xy);

179template <class Equilibrium>
180RhoE LocalEquilibriumBase<Equilibrium>::extrapolate(const Cell &cell) const {
181 if (found_equilibrium) {
182 return average(cell, [this](const XYZ &xy) { return extrapolate(xy); });

188template <class Equilibrium>
189std::string LocalEquilibriumBase<Equilibrium>::str(int verbose) const {
190 if (verbose == 0) {
191 if (found_equilibrium) {
192 return "No equilibrium found.";
193 } else {
194 return string_format("theta = %s @ x_ref = %s",
195 format_as_list(theta).c_str(),
196 format_as_list(x_ref).c_str());
197 }
198 } else if (verbose >= 1) {
199 return string_format("theta = %s @ x_ref = %s; found_equilibrium = %d",
200 format_as_list(theta).c_str(),
201 format_as_list(x_ref).c_str(),
202 found_equilibrium);
203 }
204
205 return "";
206}
