117template <class X>
118class StaticRungeKutta {
119private:
120 using function_t = std::function<X(double, const X &)>;
121
122public:
123 StaticRungeKutta(const function_t f, const std::string &method)
124 : f(f), tableau(make_tableau(method)), tendencies(tableau.n_stages) {}
125
126 X operator()(const X &x0, double t, double dt) {
127 // stage 0
128 tendencies[0] = f(t, x0);
129
130 // stages 1, ..., s
131 for (int_t stage = 1; stage < tableau.n_stages; ++stage) {
132 double tx = t + tableau.c[stage] * dt;
133
134 auto x = runge_kutta_sum(x0, tableau.a[stage], dt);
135 tendencies[stage] = f(tx, x);
136 }
137
138 return runge_kutta_sum(x0, tableau.b, dt);
139 }
140
141protected:
142 X runge_kutta_sum(X x, const array<double, 1> &coeffs, double dt) const {
143 int_t n_stages = coeffs.shape(0);
144 for (int_t stage = 0; stage < n_stages; ++stage) {
145 if (coeffs[stage] != 0.0) {
146 x += dt * coeffs[stage] * tendencies[stage];
147 }
148 }
149
150 return x;
151 }
152
153private:
154 function_t f;
155 ButcherTableau tableau;
156 std::vector<X> tendencies;
157};
