18template <class E1, class E2>
19class PointwiseSum : public PolynomialCRTP<PointwiseSum<E1, E2>> {
20public:
21 PointwiseSum(const PolynomialCRTP<E1> &e1, const PolynomialCRTP<E2> &e2)
22 : e1(static_cast<const E1 &>(e1)), e2(static_cast<const E2 &>(e2)) {
23
24 static_assert(E1::n_coeffs() == E2::n_coeffs(),
25 "Number of coefficients mismatch.");
26 static_assert(E1::n_vars() == E2::n_vars(),
27 "Number of variables mismatch.");
28 }
29
30 static constexpr int n_coeffs() { return E1::n_coeffs(); }
31 static constexpr int n_vars() { return E1::n_vars(); }
32
33 int degree() const { return zisa::max(e1.degree(), e2.degree()); }
34 int n_dims() const { return zisa::max(e1.n_dims(), e2.n_dims()); }
35
36 template <class... Indices>
37 double a(Indices &&... indices) const {
38 return this->e1.a(std::forward<Indices>(indices)...)
39 + this->e2.a(std::forward<Indices>(indices)...);
40 }
41
42 template <class... Indices>
43 double c(Indices &&... indices) const {
44 return (e1.degree() > e2.degree()
45 ? e1.c(std::forward<Indices>(indices)...)
46 : e2.c(std::forward<Indices>(indices)...));
47 }
48
49 const XYZ &x_center() const {
50 return (e1.degree() > e2.degree() ? e1.x_center() : e2.x_center());
51 }
52
53 double reference_length() const {
54 return (e1.degree() > e2.degree() ? e1.reference_length()
55 : e2.reference_length());

57
58private:

60 const E2 &e2;
