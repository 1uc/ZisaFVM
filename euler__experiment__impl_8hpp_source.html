34template <class EOS, class Gravity>
35void EulerExperiment<EOS, Gravity>::do_post_run(
36 const std::shared_ptr<AllVariables> &u1) {
37
38 if (!has_key(params, "reference")) {
39 // Post processing the reference solution is not requested.
40 return;
41 }
42
43 auto reference_solution = deduce_reference_solution(*u1);
44
45 std::vector<std::string> coarse_grid_paths
46 = params["reference"]["coarse_grids"];
47
48 auto grid_factory = choose_grid_factory();
49
50 down_sample_euler_reference(
51 *reference_solution, coarse_grid_paths, grid_factory, "reference.h5");
52
53 auto fng = choose_file_name_generator();
54 auto steady_state_filename = fng->steady_state_filename;
55 auto u_delta = std::make_shared<AllVariables>(load_serial<AllVariables>(
56 steady_state_filename, all_labels<euler_var_t>()));
57
58 for (int_t i = 0; i < u_delta->size(); ++i) {
59 (*u_delta)[i] = (*u1)[i] - (*u_delta)[i];
60 }
61
62 auto grid = choose_grid();
63 auto local_eos = choose_local_eos();
64 auto weno_params = choose_weno_reference_params();
65 auto local_rc_params = choose_local_rc_params();
66 auto rc = make_reconstruction_array<NoEquilibrium,
67 CWENO_AO,
68 UnityScaling,
69 EOS,
70 Gravity>(
71 grid, weno_params, *local_eos, gravity, local_rc_params);
72
73 auto grc = std::make_shared<
74 EulerGlobalReconstruction<NoEquilibrium, CWENO_AO, UnityScaling>>(
75 weno_params, std::move(rc));
76
77 auto delta = deduce_reference_solution_eq(*u_delta, grc);
78 down_sample_euler_reference(
79 *delta, coarse_grid_paths, grid_factory, "delta.h5");
80}

93template <class EOS, class Gravity>
94void EulerExperiment<EOS, Gravity>::do_post_process() {
95 auto fng = choose_file_name_generator();
96 auto data_filename = find_last_data_file(*fng);
97
98 auto dummy_simulation_clock = compute_simulation_clock();
99 auto sfng = std::make_shared<SingleFileNameGenerator>(data_filename);
100 auto data_source = compute_data_source(sfng);
101
102 auto all_vars = std::make_shared<AllVariables>(choose_all_variable_dims());
103 (*data_source)(*all_vars, *dummy_simulation_clock);
104
105 auto bc = choose_boundary_condition();
106 bc->apply(*all_vars, dummy_simulation_clock->current_time());
107
108 do_post_run(all_vars);
109}

122template <class EOS, class Gravity>
123std::pair<std::shared_ptr<AllVariables>, std::shared_ptr<AllVariables>>
124EulerExperiment<EOS, Gravity>::load_initial_conditions() {
125 auto simulation_clock = choose_simulation_clock();
126 auto [data_file, steady_state_file] = compute_restart_datafile();
127
128 auto fng = choose_file_name_generator();
129 fng->advance_to(data_file);
130
131 auto sfng = std::make_shared<FixedFileNameGenerator>(
132 std::vector<std::string>{data_file, steady_state_file});
133 auto data_source = compute_data_source(sfng);
134 auto all_vars = std::make_shared<AllVariables>(choose_all_variable_dims());
135 (*data_source)(*all_vars, *simulation_clock);
136
137 std::shared_ptr<AllVariables> steady_state = nullptr;
138 if (zisa::file_exists(steady_state_file)) {
139 steady_state = std::make_shared<AllVariables>(choose_all_variable_dims());
140
141 auto dummy_simulation_clock = compute_simulation_clock();
142 (*data_source)(*steady_state, *dummy_simulation_clock);
143 }
144
145 return {all_vars, steady_state};
146}

187template <class EOS, class Gravity>
188std::shared_ptr<ReferenceSolution>
189EulerExperiment<EOS, Gravity>::deduce_reference_solution(
190 const AllVariables &u1) {
191 auto local_eos = choose_local_eos();
192 auto grid = choose_grid();
193 auto weno_params = choose_weno_reference_params();
194 auto local_rc_params = choose_local_rc_params();
195
196 if (params["reference"]["equilibrium"] == "constant") {
197 LOG_WARN("No, this needs to be local");
198 auto rc = make_reconstruction_array<NoEquilibrium,
199 CWENO_AO,
200 EulerScaling<eos_t>,
201 eos_t,
202 gravity_t>(
203 grid, weno_params, *local_eos, gravity, local_rc_params);
204
205 auto grc = std::make_shared<EulerGlobalReconstruction<NoEquilibrium,
206 CWENO_AO,
207 EulerScaling<eos_t>>>(
208 weno_params, std::move(rc));
209 return deduce_reference_solution_eq(u1, std::move(grc));
210 }
211
212 if (params["reference"]["equilibrium"] == "isentropic") {
213 auto rc = make_reconstruction_array<IsentropicEquilibrium<eos_t, gravity_t>,
214 CWENO_AO,
215 EulerScaling<eos_t>,
216 eos_t,
217 gravity_t>(
218 grid, weno_params, *local_eos, gravity, local_rc_params);
219 auto grc = std::make_shared<
220 EulerGlobalReconstruction<IsentropicEquilibrium<eos_t, gravity_t>,
221 CWENO_AO,
222 EulerScaling<eos_t>>>(weno_params,
223 std::move(rc));
224
225 return deduce_reference_solution_eq(u1, grc);
226 }
227
228 LOG_ERR("Failed to deduce reference solutions.");
229}

276template <class EOS, class Gravity>
277std::shared_ptr<RateOfChange> EulerExperiment<EOS, Gravity>::choose_flux_bc() {
278 std::string flux_bc = params["flux-bc"]["mode"];
279 auto grid = choose_grid();
280 auto local_eos = choose_local_eos();
281
282 if (flux_bc == "none") {
283 return std::make_shared<NoFluxBC>();
284 }
285
286 if (flux_bc == "constant") {
287 return std::make_shared<FluxBC<eos_t>>(euler, local_eos, grid);
288 }
289
290 if (flux_bc == "isentropic") {
291 auto qr = choose_edge_rule();
292 using eq_t = IsentropicEquilibrium<eos_t, gravity_t>;
293
294 return std::make_shared<EquilibriumFluxBC<eq_t, EOS, Gravity>>(
295 euler, local_eos, gravity, grid, qr);
296 }
297
298 LOG_ERR(
299 string_format("Unknown flux boundary condition. [%s]", flux_bc.c_str()));
300}

439 rc);
