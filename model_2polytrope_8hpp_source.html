76template <class EOS>
77AllVariables make_general_polytrope_profile(const Grid &grid,
78 const EOS &eos,
79 const PolytropeParams &params) {
80
81 auto spacing = zisa::linear_spacing(0.0, 1.0, 2048);
82 auto [r, rho_mass] = compute_polytrope_profile(params, spacing);
83 auto interpolate = NonUniformLinearInterpolation<Cartesian<2>>(r, rho_mass);
84
85 double K = params.K_center;
86
87 auto cvars = zisa::cell_averages(
88 [K, &eos, &interpolate](const XYZ &x) {
89 auto rho_mass = interpolate(zisa::norm(x));
90 double rho = rho_mass[0];
91 auto rhoK = RhoEntropy{rho, K};
92 return euler_var_t{rho, 0.0, 0.0, 0.0, eos.internal_energy(rhoK)};
93 },
94 grid);
95
96 return AllVariables{std::move(cvars), GridVariables{}};
97}
