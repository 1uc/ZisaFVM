15template <class EOS, class Gravity>
16struct IsentropicEquilibrium {
17public:
18 using equilibrium_values_t = isentropic_equilibrium_values_t<EOS>;
19 using xvars_t = euler_xvar_t;
20
21public:
22 IsentropicEquilibrium() = default;
23 explicit IsentropicEquilibrium(std::shared_ptr<EOS> eos,
24 std::shared_ptr<Gravity> gravity)
25 : eos(std::move(eos)), gravity(std::move(gravity)) {}
26
27 RhoE extrapolate(const equilibrium_values_t &theta_ref,
28 const XYZ &x_ref,
29 const XYZ &x) const {
30
31 auto theta = extrapolate_theta(theta_ref, x_ref, x);
32 return conserved_variables(*eos, theta);
33 }
34
35 std::pair<RhoE, xvars_t> extrapolate_full(equilibrium_values_t theta_ref,
36 const XYZ &x_ref,
37 const XYZ &x) const {
38
39 auto theta = extrapolate_theta(theta_ref, x_ref, x);
40 return full_variables(*eos, theta);
41 }
42
43private:
44 equilibrium_values_t extrapolate_theta(equilibrium_values_t theta_ref,
45 const XYZ &x_ref,
46 const XYZ &x) const {
47
48 double phi_ref = gravity->phi(x_ref);
49 double phi = gravity->phi(x);
50
51 auto h_ref = theta_ref.h();
52 auto s_ref = theta_ref.s();
53
54 double h = h_ref + phi_ref - phi;
55 double s = s_ref;
56
57 // This awkwardness serves a purpose, because `equilibrium_variables_t`
58 // can include more than {h, s}, e.g. `rhoT_guess`.
59 theta_ref.h() = h;
60 theta_ref.s() = s;
61
62 return theta_ref;
63 }
64
65public: // TODO why are these public? Probably shouldn't be.
66 std::shared_ptr<EOS> eos;
67 std::shared_ptr<Gravity> gravity;
68};
