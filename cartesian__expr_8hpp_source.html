183template <class E1, class E2, class T>
184class CartesianCross : public CartesianExpr<CartesianCross<E1, E2, T>, T> {
185public:
186 using scalar_t = T;
187
188 ANY_DEVICE_INLINE CartesianCross(const CartesianExpr<E1, T> &e1,
189 const CartesianExpr<E2, T> &e2)
190 : e1(static_cast<const E1 &>(e1)), e2(static_cast<const E2 &>(e2)) {}
191
192 ANY_DEVICE_INLINE scalar_t operator()(int_t i) const {
193 switch (i) {
194 case 0:
195 return e1(1) * e2(2) - e1(2) * e2(1);
196 case 1:
197 return e1(2) * e2(0) - e1(0) * e2(2);
198 case 2:
199 return e1(0) * e2(1) - e1(1) * e2(0);
200 default:
201 return -42.0;
202 }
203 }
204
205 ANY_DEVICE_INLINE constexpr static int_t size(void) {
206 static_assert(E1::size() == 3, "Cross-product only works for 3-vectors.");
207 static_assert(E2::size() == 3, "Cross-product only works for 3-vectors.");
208
209 return E1::size();
210 }
211
212private:
213 const E1 &e1;
214 const E2 &e2;
215};

305template <class E1, class E2, class T>
306ANY_DEVICE_INLINE double dot(const CartesianExpr<E1, T> &e_,
307 const CartesianExpr<E2, T> &f_) {
308 const E1 &e = static_cast<const E1 &>(e_);
309 const E2 &f = static_cast<const E2 &>(f_);
310
311 double ef = 0;
312 for (int_t i = 0; i < e.size(); ++i) {
313 ef += e(i) * f(i);
314 }
315
316 return ef;
317}

361template <class E, class T>
362typename E::scalar_t minimum(const CartesianExpr<E, T> &e_) {
363 const E &e = static_cast<const E &>(e_);
364
365 typename E::scalar_t gbl_min = e(0);
366 for (int_t i = 1; i < e.size(); ++i) {
367 gbl_min = std::min(gbl_min, e(i));
368 }
369
370 return gbl_min;
371}

582template <class E>
583bool all(const CartesianLogic<E> &e_) {
584 const E &e = static_cast<const E &>(e_);
585
586 for (int_t i = 0; i < e.size(); ++i) {
587 if (!e(i)) {
588 return false;
589 }
590 }
591
592 return true;
593}
