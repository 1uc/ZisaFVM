223template <class E1, class E2, class E3>
224double cos_angle(const CartesianExpr<E1, double> &e1_,
225 const CartesianExpr<E2, double> &e2_,
226 const CartesianExpr<E3, double> &e3_) {
227
228 const auto &e1 = static_cast<const E1 &>(e1_);
229 const auto &e2 = static_cast<const E2 &>(e2_);
230 const auto &e3 = static_cast<const E3 &>(e3_);
231
232 return zisa::dot(normalize(e1 - e2), normalize(e3 - e2));
233}

235template <class E>
236bool isreal(const CartesianExpr<E, double> &e_) {
237 const E &e = static_cast<const E &>(e_);
238
239 for (int_t k = 0; k < e.size(); ++k) {
240 if (!isreal(e[k])) {
241 return false;
242 }
243 }
244
245 return true;
246}
