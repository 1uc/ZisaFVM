33 HelmholtzEOS(const array_const_view<double, 1> &mass_mixing_ratio,
34 const array_const_view<double, 1> &mass_number,
35 const array_const_view<double, 1> &charge_number) {
36
37 int status = -1;
38 int n_species = integer_cast<int>(charge_number.size());
39 composition_bar_c(raw_ptr(mass_mixing_ratio),
40 raw_ptr(mass_number),
41 raw_ptr(charge_number),
42 &n_species,
43 &a_bar,
44 &z_bar,
45 &status);
46
47 LOG_ERR_IF(status != 0 || !ispositive(a_bar) || !ispositive(z_bar),
48 string_format("EOS failed: status = %d, a_bar = %e, z_bar = %e, "
49 "X = %s, A = %s, Z = %s",
50 status,
51 a_bar,
52 z_bar,
53 format_as_list(mass_mixing_ratio).c_str(),
54 format_as_list(mass_number).c_str(),
55 format_as_list(charge_number).c_str()));
56 }

104 euler_full_xvars_t
105 full_extra_variables(int id, double tv1, double tv2) const {
106 int status = -1;
107 eos_state_type eos_ret;
108
109 LOG_ERR_IF(
110 (tv1 <= 0.0) || (tv2 <= 0.0),
111 string_format(
112 "Invalid input to EOS. %d : tv1 = %e, tv2 = %e", id, tv1, tv2));
113
114 helmholtz_eos_bar_c(&id, &tv1, &tv2, &a_bar, &z_bar, &eos_ret, &status);
115 LOG_ERR_IF(
116 status != 0,
117 string_format("EOS failed. %d [%d, %e, %e] a_bar = %e, z_bar = %e",
118 status,
119 id,
120 tv1,
121 tv2,
122 a_bar,
123 z_bar));
124
125 return full_extra_variables(eos_ret);
126 }
127
128 euler_full_xvars_t full_extra_variables(const EnthalpyEntropy &theta,
129 const RhoT &rhoT_guess) const {
130
131 int status = -1;
132 eos_state_type ret;
133
134 double h = theta.h();
135 double s = theta.s();
136
137 double rho_guess = rhoT_guess.rho();
138 double T_guess = rhoT_guess.T();
139
140 LOG_ERR_IF(
141 ((h <= 0.0) || (s <= 0.0) || (rho_guess <= 0.0) || (T_guess <= 0.0)),
142 string_format("Invalid input to EOS. h = %e, s = %e; rho_guess "
143 "= %e, T_guess = %e",
144 h,
145 s,
146 rho_guess,
147 T_guess));
148
149 helmholtz_eos_wguess_bar_c(
150 &TD_HS, &h, &s, &a_bar, &z_bar, &ret, &status, &T_guess, &rho_guess);
151
152 LOG_ERR_IF(status != 0,
153 string_format("EOS TD_HS failed. %d, [%.3e, %.3e; (%.3e, %.3e)] "
154 "a_bar = %e, z_bar = %e",
155 status,
156 h,
157 s,
158 rho_guess,
159 T_guess,
160 a_bar,
161 z_bar));
162
163 return full_extra_variables(ret);
164 }

166 euler_full_xvars_t full_extra_variables(const eos_state_type &eos_ret) const {
167
168 euler_full_xvars_t ret;
169 ret.rho = eos_ret.d;
170 ret.E = eos_ret.d * eos_ret.e;
171 ret.a = eos_ret.cs;
172 ret.h = eos_ret.h;
173 ret.s = eos_ret.s;
174 ret.p = eos_ret.p;
175 ret.T = eos_ret.t;
176
177 return ret;
178 }
