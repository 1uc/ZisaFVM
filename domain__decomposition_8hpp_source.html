58template <class T>
59array<T, 1> extract_subarray(const array<T, 1> &global_array,
60 const array<int_t, 1> &global_indices) {
61 auto n_cells_local = global_indices.size();
62
63 auto local_array = array<T, 1>(n_cells_local);
64 for_each(index_range(n_cells_local),
65 [&](int_t i) { local_array(i) = global_array(global_indices(i)); });
66
67 return local_array;
68}
