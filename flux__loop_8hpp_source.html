22class FluxLoop : public RateOfChange {
23protected:
24 using cvars_t = typename Model::cvars_t;

28public:
29 FluxLoop(std::shared_ptr<Grid> grid_,
30 std::shared_ptr<Model> model_,
31 std::shared_ptr<LEOS> local_eos_,
32 std::shared_ptr<grc_t> global_reconstruction_,
33 std::shared_ptr<HaloExchange> halo_exchange_,
34 EdgeRule edge_rule)
35 : grid(std::move(grid_)),
36 model(std::move(model_)),
37 local_eos(std::move(local_eos_)),
38 global_reconstruction(std::move(global_reconstruction_)),
39 edge_rule(std::move(edge_rule)),
40 halo_exchange(std::move(halo_exchange_)) {
41
42 avar_flux_allocator
43 = std::make_shared<block_allocator<array<double, 1>>>(128);
44
45 interior_cells.reserve(this->grid->n_cells);
46 exterior_cells.reserve(this->grid->n_cells / 2);
47
48 for (int_t i = 0; i < this->grid->n_cells; ++i) {
49 const auto &s = this->global_reconstruction->stencil(i);
50 for (int_t j : s) {
51 if (this->grid->cell_flags[j].ghost_cell) {
52 exterior_cells.push_back(i);
53 break;
54 }
55 }
56
57 if (exterior_cells.empty() || exterior_cells.back() != i) {
58 interior_cells.push_back(i);
59 }
60 }
61
62 std::sort(interior_cells.begin(), interior_cells.end());
63 std::sort(exterior_cells.begin(), exterior_cells.end());
64
65 interior_faces.reserve(this->grid->n_interior_edges);
66 exterior_faces.reserve((this->grid->n_edges - this->grid->n_interior_edges)
67 * 2);
68
69 // In the context of a grid, any edge which has two neighbours is an
70 // interior edge. Here we attempt to find the edges which are safe to update
71 // before the halo has been exchanged.
72 for (int_t e = 0; e < this->grid->n_interior_edges; ++e) {
73 auto [iL, iR] = this->grid->left_right(e);
74
75 auto is_interior
76 = !contains(exterior_cells, iL) && !contains(exterior_cells, iR);
77
78 if (is_interior) {
79 interior_faces.push_back(e);
80
81 } else {
82 auto is_left_ghost = this->grid->cell_flags[iL].ghost_cell;
83 auto is_right_ghost = this->grid->cell_flags[iR].ghost_cell;
84
85 if (!(is_left_ghost && is_right_ghost)) {
86 exterior_faces.push_back(e);
87 }
88 }
89 }
90 }

92 bool contains(const std::vector<int_t> &cells, int_t i) const {
93 return std::lower_bound(cells.begin(), cells.end(), i) != cells.end();
94 }

96 virtual void compute(AllVariables &tendency,
97 const AllVariables &current_state,
98 double /* t */) const override {
99
100 (*halo_exchange)(const_cast<AllVariables &>(current_state));
101 compute_patch(tendency, current_state, interior_cells, interior_faces);
102 (*halo_exchange).wait();
103 compute_patch(tendency, current_state, exterior_cells, exterior_faces);
104 }

106 void compute_patch(AllVariables &tendency,
107 const AllVariables &current_state,
108 const array_const_view<int_t, 1> &cells,
109 const array_const_view<int_t, 1> &edges) const {
110
111 (*local_eos).compute(current_state, cells);
112 (*global_reconstruction).compute(current_state, cells);
113
114 const auto n_avars = tendency.avars.shape(1);
115 const auto n_interior_edges = edges.size();
116#if ZISA_HAS_OPENMP == 1
117#pragma omp parallel
118#endif
119 {
120 auto qnf_guard = fetch_avars_flux_buffer(n_avars);
121 auto &qnf = *qnf_guard;
122
123#if ZISA_HAS_OPENMP == 1
124#pragma omp for ZISA_OMP_FOR_SCHEDULE_DEFAULT
125#endif
126 for (int_t ie = 0; ie < n_interior_edges; ++ie) {
127 auto e = edges[ie];
128
129 const auto &face = grid->faces(e);
130
131 int_t iL, iR;
132 std::tie(iL, iR) = grid->left_right(e);
133 const auto &eosL = *(*local_eos)(iL);
134 const auto &eosR = *(*local_eos)(iR);
135
136 auto rc = [this, &face = face](int_t i, const XYZ &x) {
137 auto u = cvars_t((*global_reconstruction)(i)(x));
138 coord_transform(u, face);
139
140 return u;
141 };
142
143 auto nf = cvars_t::zeros();
144 zisa::fill(qnf, 0.0);
145
146 const auto n_qr = face.qr.weights.size();
147 for (int_t k = 0; k < n_qr; ++k) {
148 const auto w = face.qr.weights[k];
149 const auto x = face.qr.points[k];
150
151 const auto uL = rc(iL, x);
152 const auto uR = rc(iR, x);
153
154 const auto [f, speeds] = numerical_flux(eosL, uL, eosR, uR);
155 nf += w * f;
156
157 for (int_t a = 0; a < n_avars; ++a) {
158 const auto qL = (*global_reconstruction)(iL).tracer(x, a);
159 const auto qR = (*global_reconstruction)(iR).tracer(x, a);
160 qnf[a] += w * tracer_flux(uL, uR, qL, qR, speeds);
161 }
162 }
163
164 inv_coord_transform(nf, face);
165
166 for (int_t k = 0; k < cvars_t::size(); ++k) {
167 auto nfL = nf(k) / grid->volumes(iL);
168#if ZISA_HAS_OPENMP == 1
169#pragma omp atomic
170#endif
171 tendency.cvars(iL, k) -= nfL;
172
173 auto nfR = nf(k) / grid->volumes(iR);
174#if ZISA_HAS_OPENMP == 1
175#pragma omp atomic
176#endif
177 tendency.cvars(iR, k) += nfR;
178 }
179
180 for (int_t k = 0; k < n_avars; ++k) {
181 const auto qfL = qnf(k) / grid->volumes(iL);
182#if ZISA_HAS_OPENMP == 1
183#pragma omp atomic
184#endif
185 tendency.avars(iL, k) -= qfL;
186
187 const auto qfR = qnf(k) / grid->volumes(iR);
188#if ZISA_HAS_OPENMP == 1
189#pragma omp atomic
190#endif
191 tendency.avars(iR, k) += qfR;
192 }
193 }
194 }
195 }
