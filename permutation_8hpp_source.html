72template <class T>
73void apply_cycle(array_view<T, 1> &data, const Cycle &cycle) {
74 T first = std::move(data[cycle[0]]);
75
76 for (int_t i = 0; i < cycle.size() - 1; ++i) {
77 data[cycle[i]] = std::move(data[cycle[i + 1]]);
78 }
79 data[cycle[cycle.size() - 1]] = std::move(first);
80}

82template <class T>
83void apply_cycle(array_view<T, 2> &data, const Cycle &cycle) {
84 auto cycle_length = cycle.size();
85 auto n_vars = data.shape(1);
86
87 array<T, 1> first(n_vars);
88 for (int_t k = 0; k < n_vars; ++k) {
89 first(k) = std::move(data(cycle(0), k));
90 }
91
92 for (int_t i = 0; i < cycle_length - 1; ++i) {
93 for (int_t k = 0; k < n_vars; ++k) {
94 data(cycle(i), k) = std::move(data(cycle(i + 1), k));
95 }
96 }
97
98 for (int_t k = 0; k < n_vars; ++k) {
99 data(cycle(cycle_length - 1), k) = std::move(first(k));
100 }
101}

103template <class T, int n_dims>
104void apply_permutation(array_view<T, n_dims> data, const Permutation &sigma) {
105 if(data.shape(0) == 0) {

116template <class T>
117void reverse_cycle(array_view<T, 1> &data, const Cycle &cycle) {
118 auto cycle_size = cycle.size();
119 T last = std::move(data[cycle[cycle_size-1]]);
120
121 for (int_t i = cycle_size-1; i > 0; --i) {
122 data[cycle[i]] = std::move(data[cycle[i-1]]);
123 }
124 data[cycle[0]] = std::move(last);
125}

127template <class T>
128void reverse_cycle(array_view<T, 2> &data, const Cycle &cycle) {
129 auto cycle_length = cycle.size();
130 auto n_vars = data.shape(1);
131
132 array<T, 1> last(n_vars);
133 for (int_t k = 0; k < n_vars; ++k) {
134 last(k) = std::move(data(cycle(cycle_length-1), k));
135 }
136
137 for (int_t i = cycle_length-1; i > 0; --i) {
138 for (int_t k = 0; k < n_vars; ++k) {
139 data(cycle(i), k) = std::move(data(cycle(i-1), k));
140 }
141 }
142
143 for (int_t k = 0; k < n_vars; ++k) {
144 data(cycle(0), k) = std::move(last(k));
145 }
146}
