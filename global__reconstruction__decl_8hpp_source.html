76template <class Equilibrium, class RC, class Scaling, class EOS, class Gravity>
77array<LocalReconstruction<Equilibrium, RC, Scaling>, 1>
78make_reconstruction_array(const std::shared_ptr<Grid> &grid,
79 const HybridWENOParams &weno_params,
80 const LocalEOSState<EOS> &local_eos_state,
81 const std::shared_ptr<Gravity> &gravity,
82 const LocalRCParams &local_rc_params) {
83
84 auto n_cells = grid->n_cells;
85
86 auto lrc = array<LocalReconstruction<Equilibrium, RC, Scaling>, 1>(
87 shape_t<1>{n_cells});
88 for (int_t i_cell = 0; i_cell < n_cells; ++i_cell) {
89 const auto &eos = local_eos_state(i_cell);
90
91 auto rc = RC(grid, i_cell, weno_params);
92 lrc[i_cell] = LocalReconstruction<Equilibrium, RC, Scaling>(
93 grid,
94 LocalEquilibrium<Equilibrium>(
95 make_equilibrium<Equilibrium>(eos, gravity)),
96 rc,
97 i_cell,
98 Scaling(eos),
99 local_rc_params);
100 }
101
102 return lrc;
103}

105template <class Equilibrium, class RC, class Scaling, class EOS, class Gravity>
106array<LocalReconstruction<Equilibrium, RC, Scaling>, 1>
107make_reconstruction_array(const std::shared_ptr<Grid> &grid,
108 const array<StencilFamily, 1> &stencils,
109 const HybridWENOParams &weno_params,
110 const LocalEOSState<EOS> &local_eos_state,
111 const std::shared_ptr<Gravity> &gravity,
112 const LocalRCParams &local_rc_params) {
113
114 auto n_cells = grid->n_cells;
115
116 auto lrc = array<LocalReconstruction<Equilibrium, RC, Scaling>, 1>(
117 shape_t<1>{n_cells});
118 for (int_t i = 0; i < n_cells; ++i) {
119 const auto &eos = local_eos_state(i);
120 auto eq = make_equilibrium<Equilibrium>(eos, gravity);
121 auto scaling = Scaling(eos);
122
123 auto o1_params = HybridWENOParams(
124 {{1}, {"c"}, {1.0}}, {1.0}, weno_params.epsilon, weno_params.exponent);
125
126 if (stencils[i].size() == 1 && stencils[i].order() == 1) {
127 lrc[i] = LocalReconstruction<Equilibrium, RC, Scaling>(
128 grid,
129 LocalEquilibrium(eq),
130 RC(grid, stencils[i], i, o1_params),
131 i,
132 scaling,
133 local_rc_params);
134
135 } else {
136 lrc[i] = LocalReconstruction<Equilibrium, RC, Scaling>(
137 grid,
138 LocalEquilibrium(eq),
139 RC(grid, stencils[i], i, weno_params),
140 i,
141 scaling,
142 local_rc_params);
143 }
144 }
145
146 return lrc;
147}
