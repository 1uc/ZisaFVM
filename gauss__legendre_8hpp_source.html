57 template <int n>
58 void quadrature_points(GaussLegendre<n> &gl,
59 const FourierSeries<n> &p,
60 const FourierSeriesDerivative<n> &dp) {
61 bool is_even = n % 2 == 0;
62
63 if (is_even) {
64 quadrature_points_even(gl, p, dp);
65 } else {
66 quadrature_points_odd(gl, p, dp);
67 }

70 template <int n>
71 void quadrature_points_odd(GaussLegendre<n> &gl,
72 const FourierSeries<n> &p,
73 const FourierSeriesDerivative<n> &dp) {
74 gl.points[n / 2] = zisa::pi / 2;
75
76 for (int k = 1; k <= n / 2; ++k) {
77 double theta_guess
78 = (k < 2 ? zisa::pi / 2 - k * zisa::pi / n
79 : 2.0 * gl.points[n / 2 + k - 1] - gl.points[n / 2 + k - 2]);
80
81 gl.points[n / 2 + k] = newton(p, dp, theta_guess, 1e-12);
82 }
83 }
84
85 template <int n>
86 void quadrature_points_even(GaussLegendre<n> &gl,
87 const FourierSeries<n> &p,
88 const FourierSeriesDerivative<n> &dp) {
89 for (int k = 0; k < n / 2; ++k) {
90 double theta_guess
91 = (k < 2 ? zisa::pi / 2 - (k + 0.5) * zisa::pi / n
92 : 2.0 * gl.points[n / 2 + k - 1] - gl.points[n / 2 + k - 2]);
93
94 gl.points[n / 2 + k] = newton(p, dp, theta_guess, 1e-12);
95 }
96 }

98 template <int n>
99 void quadrature_weights(GaussLegendre<n> &gl,
100 const FourierSeries<n> &,
101 const FourierSeriesDerivative<n> &dp) {
102 bool is_even = n % 2 == 0;
103 bool is_odd = !is_even;
104
105 if (is_odd) {
106 gl.weights[n / 2] = weight(dp, gl.points[n / 2]);
107 }
108
109 for (int k = 0; k < n / 2; ++k) {
110 int i_upper = (is_even ? n / 2 + k : n / 2 + k + 1);
111 int i_lower = n / 2 - 1 - k;
112
113 gl.weights[i_upper] = weight(dp, gl.points[i_upper]);
114 gl.weights[i_lower] = gl.weights[i_upper];
115 }
116 }

118 template <int n>
119 double weight(const FourierSeriesDerivative<n> &dp, double theta) {
120 return (2 * n + 1) / zisa::pow<2>(dp(theta));
121 }

123 template <int n>
124 void convert_theta_to_x(GaussLegendre<n> &gl) {
125 bool is_even = n % 2 == 0;
126 bool is_odd = !is_even;
127
128 if (is_odd) {
129 gl.points[n / 2] = zisa::cos(gl.points[n / 2]);
130 }
131
132 for (int k = 0; k < n / 2; ++k) {
133 int i_upper = (is_even ? n / 2 + k : n / 2 + k + 1);
134 int i_lower = n / 2 - 1 - k;
135
136 gl.points[i_upper] = zisa::cos(gl.points[i_upper]);
137 gl.points[i_lower] = -gl.points[i_upper];
138 }
139 }

141 template <int n>
142 FourierSeries<n> fourier_poly() {
143 constexpr bool is_even = n % 2 == 0;
144
145 FourierSeries<n> p;
146 for (int i = 0; i <= n; ++i) {
147 p.at(i) = 0.0;
148 }
149
150 p.at(n) = a_nn(n);
151 for (int l = 2; l <= n; l += 2) {
152 double factor
153 = double((l - 1) * (2 * n - l + 2)) / double(l * (2 * n - l + 1));
154 p.at(n - l) = factor * p.at(n - l + 2);
155 }
156
157 if (is_even) {
158 p.at(0) *= 0.5;
159 }
160
161 return p;
162 }
163
164protected:
165 double a_nn(int n) {
166 double a = zisa::sqrt(2.0);
167
168 for (int i = 1; i <= n; ++i) {
169 a = zisa::sqrt(1.0 - 1.0 / (4.0 * zisa::pow<2>(i))) * a;
170 }
171
172 return a;
173 }
174};
