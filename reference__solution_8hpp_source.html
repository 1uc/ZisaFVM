25template <class Equilibrium, class Scaling>
26class EulerReferenceSolution : public ReferenceSolution {
27protected:
28 using eq_t = Equilibrium;
29 using grc_t = EulerGlobalReconstruction<Equilibrium, CWENO_AO, Scaling>;

31public:
32 EulerReferenceSolution(std::shared_ptr<Grid> fine_grid_,
33 const AllVariables &fine_vars,
34 std::shared_ptr<grc_t> grc_)
35 : fine_grid(std::move(fine_grid_)),
36 grc(std::move(grc_)),
37 n_cvars(fine_vars.cvars.shape(1)) {
38
39 // FIXME we now have tracers.
40 LOG_ERR_IF(fine_vars.dims().n_avars != n_avars, "Dimension mismatch.");
41 grc->compute(fine_vars);
42 }

44 virtual std::shared_ptr<AllVariables>
45 average(const Grid &coarse_grid) const override {
46 auto n_cells = coarse_grid.n_cells;
47
48 auto ref = std::make_shared<AllVariables>(
49 AllVariablesDimensions{n_cells, n_cvars, n_avars});
50
51 auto &u_coarse = ref->cvars;
52
53 {
54 int_t i_guess = 0;
55
56 for (int_t i = 0; i < n_cells; ++i) {
57 if (coarse_grid.cell_flags(i).ghost_cell) {
58 for (int_t k_var = 0; k_var < n_cvars; ++k_var) {
59 u_coarse(i, k_var) = 0.0;
60 }
61 } else {
62 const auto &cell = coarse_grid.cells(i);
63 auto u_bar = cvars_average(cell, i_guess);
64 for (int_t k_var = 0; k_var < n_cvars; ++k_var) {
65 u_coarse(i, k_var) = u_bar[k_var];
66 }
67 }
68 }
69 }
70
71 return ref;
72 }
73
74 double q_ref(const XYZ &x, int_t k, int_t &i_guess) const override {
75 auto i_cell = locate(*fine_grid, x, i_guess);
76 LOG_ERR_IF(!i_cell,
77 string_format("Failed to locate the cell. x = %s",
78 format_as_list(x).c_str()));
79
80 i_guess = *i_cell;
81 return (*grc)(*i_cell)(x)[k];
82 }
83
84protected:
85 euler_var_t cvars_average(const Cell &cell, int_t &i_guess) const {
86 auto f = [this, &i_guess](const XYZ &x) { return u_ref(x, i_guess); };
87 return zisa::average(cell, f);
88 }
89
90 euler_var_t u_ref(const XYZ &x, int_t &i_guess) const {
91 auto i_cell = locate(*fine_grid, x, i_guess);
92 LOG_ERR_IF(!i_cell,
93 string_format("Failed to locate the cell. x = %s",
94 format_as_list(x).c_str()));
95
96 i_guess = *i_cell;
97
98 return euler_var_t((*grc)(*i_cell)(x));
99 }
100
101 HybridWENOParams weno_params() {
102 if (fine_grid->n_dims() == 2) {
103 return HybridWENOParams{
104 {{5, 2, 2, 2}, {"c", "b", "b", "b"}, {2.0, 1.5, 1.5, 1.5}},
105 {100.0, 1.0, 1.0, 1.0},
106 1e-10,
107 4.0};
108 }
109
110 if (fine_grid->n_dims() == 3) {
111 return HybridWENOParams{{{4, 2, 2, 2, 2},
112 {"c", "b", "b", "b", "b"},
113 {3.0, 2.0, 2.0, 2.0, 2.0}},
114 {100.0, 1.0, 1.0, 1.0, 1.0},
115 1e-10,
116 4.0};
117 }
118
119 LOG_ERR("Implement first.");
120 }
121
122private:
123 std::shared_ptr<Grid> fine_grid;
124 std::shared_ptr<grc_t> grc;
125
126 int_t n_cvars;
127 int_t n_avars = 0;
128
129 mutable int_t last_index_ = 0;
130};
