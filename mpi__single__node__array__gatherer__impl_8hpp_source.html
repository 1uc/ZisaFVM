32 template <class T, int n_dims>
33 void MPISingleNodeArrayGatherer<T, n_dims>::receive(
34  const MPISingleNodeArrayGatherer::view_t &view) const {
35  std::vector<zisa::mpi::Request> requests;
36  requests.reserve(integer_cast<size_t>(comm_size));
37 
38  for (int sender_rank = 0; sender_rank < comm_size; ++sender_rank) {
39  if (sender_rank != rank) {
40  auto i0 = (*array_info).partition[sender_rank];
41  auto i1 = (*array_info).partition[sender_rank + 1];
42  auto sub_array = slice(view, i0, i1);
43 
44  auto request = zisa::mpi::irecv(sub_array, sender_rank, tag, comm);
45  requests.push_back(std::move(request));
46  }
47  }
48 
49  zisa::mpi::wait_all(requests);
50 }
