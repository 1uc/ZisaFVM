100 template <class Equilibrium, class RC, class Scaling>
101 void EulerGlobalReconstruction<Equilibrium, RC, Scaling>::compute(
102  const AllVariables &current_state) {
103  auto n_cells = current_state.cvars.shape(0);
104 
105 #if ZISA_HAS_OPENMP == 1
106 #pragma omp parallel
107 #endif
108  {
109  auto qbar_local = qbar_allocator->allocate(shape_t<1>{max_stencil_size});
110  auto tracer_local = tracer_allocator->allocate(
111  shape_t<2>{max_stencil_size, current_state.avars.shape(1)});
112  auto polys = polys_allocator->allocate(shape_t<1>{n_polys});
113  auto rhs = rhs_allocator->allocate(
114  shape_t<2>{max_stencil_size, WENOPoly::n_vars()});
115 
116 #if ZISA_HAS_OPENMP == 1
117 #pragma omp for ZISA_OMP_FOR_SCHEDULE_DEFAULT
118 #endif
119  for (int_t i = 0; i < n_cells; ++i) {
120  set_qbar_local(*qbar_local, current_state, i);
121  rc[i].compute(*rhs, *polys, *qbar_local);
122 
123  auto tracer_polys = array_view<ScalarPoly, 1>(
124  shape_t<1>(polys->shape(0)), (ScalarPoly *)(polys->raw()));
125  set_tracer_local(*tracer_local, current_state, i);
126  rc[i].compute_tracer(*rhs, tracer_polys, *tracer_local);
127  }
128  }
129 }
