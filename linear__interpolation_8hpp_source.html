20 T operator()(double r) const {
21 int_t i = index(r);
22 double alpha = (r - points(i)) / (points(i + 1) - points(i));
23 return T((1 - alpha) * values[i] + alpha * values[i + 1]);
24 }
25
26 T derivative(double r) const {

28 return T((values[i + 1] - values[i]) / (points(i + 1) - points(i)));
29 }
30
31 int_t index(double r) const {
32 auto first_past = std::lower_bound(points.cbegin(), points.cend(), r);
33
34 auto i = int_t(first_past - points.begin());
35 if (i == 0) {
36 return i;
37 }
38 assert(i < values.size());
39 return i - 1;
40 }
41
42public:
