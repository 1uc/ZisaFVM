14template <class Range, class Transform>
15auto sum(omp_policy, const Range &range, const Transform &transform) {
16 int_t i_start = range.start_index();
17 int_t i_end = range.end_index();
18
19 using ret_type = detail::return_t<Range, Transform>;
20 ret_type ret = 0;
21
22#if ZISA_HAS_OPENMP == 1

24#endif
25 for (int_t i = i_start; i < i_end; ++i) {
26 if constexpr (range_traits<Range>::has_item) {
