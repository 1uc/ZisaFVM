11template <int NDIMS>
12constexpr ANY_DEVICE_INLINE int_t poly_dof(int deg) {
13 if constexpr (NDIMS == 2) {
14 return int_t(((deg + 1) * (deg + 2)) / 2);
15 } else {
16 return int_t(((deg + 1) * (deg + 2) * (deg + 3)) / 6);
17 }
18}

20ANY_DEVICE int_t poly_dof(int deg, int n_dims);

22template <int NDIMS>
23int poly_degree(int_t n_coeffs) {
24
25 LOG_ERR_IF(n_coeffs == 0, "Too few coefficients.");
26
27 for (int d = 1; true; ++d) {
28 if (poly_dof<NDIMS>(d) > n_coeffs) {
29 return d - 1;
30 }
31 }
32}

34constexpr ANY_DEVICE_INLINE int_t poly_index(int a, int b) {
35 auto n = int_t(a) + int_t(b);
36 return int_t(((n + 1) * n) / 2 + int_t(b));
37}

39constexpr ANY_DEVICE_INLINE int_t poly_index(int a, int b, int c) {
40 return poly_dof<3>(a + b + c - 1) + poly_index(b, c);
41}

124template <int NCOEFFS, int NVARS>
125PolyND<NCOEFFS, NVARS>::PolyND(
126 const std::initializer_list<double> &coeffs_list,
127 const std::initializer_list<double> &moments_list,
128 const XYZ &x_center,
129 double reference_length,
130 int n_dims)
131 : degree_(poly_degree(coeffs_list.size(), n_dims)),
132 n_dims_(n_dims),
133 x_center_(x_center),
134 reference_length_(reference_length) {
135
136 assert(n_vars() == 1);
137 assert(coeffs_list.size() <= n_coeffs());
138 assert(coeffs_list.size() == moments_list.size());
139 assert(poly_dof(degree(), n_dims) == coeffs_list.size());
140
141 std::fill(coeffs, coeffs + n_vars() * n_coeffs(), 0.0);
142 std::copy(coeffs_list.begin(), coeffs_list.end(), coeffs);
143
144 std::fill(moments, moments + n_coeffs(), 0.0);
145 std::copy(moments_list.begin(), moments_list.end(), moments);
146}

163template <int NCOEFFS, int NVARS>
164template <class E>
165void PolyND<NCOEFFS, NVARS>::deep_copy(const PolynomialCRTP<E> &e_) {
166 const E &e = static_cast<const E &>(e_);
167
168 static_assert(E::n_coeffs() == NCOEFFS, "Number of coefficients mismatch.");
169 static_assert(E::n_vars() == NVARS, "Number of variables mismatch.");
170
171 degree_ = e.degree();
172
173 constexpr int_t n = n_coeffs() * n_vars();
174 for (int_t i = 0; i < n; ++i) {
175 coeffs[i] = e.a(i);
176 }
177
178 for (int_t i = 0; i < n_coeffs(); ++i) {
179 moments[i] = e.c(i);
180 }
181
182 x_center_ = e.x_center();
183 reference_length_ = e.reference_length();
184 n_dims_ = e.n_dims();
185}

271template <int MAX_DEGREE, int NVARS>
272Cartesian<NVARS> PolyND<MAX_DEGREE, NVARS>::eval_2d(const XYZ &xy) const {
273 auto [x, y, _] = XYZ((xy - this->x_center()) / this->reference_length());
274
275 auto d = this->degree();
276
277 Cartesian<NVARS> px(0.0);
278 double pow_x_kx = 1.0;
279 for (int kx = 0; kx <= d; ++kx) {
280 double pow_y_ky = 1.0;
281 for (int ky = 0; ky <= d - kx; ++ky) {
282 int_t i = idx(kx, ky);
283 for (int_t k = 0; k < NVARS; ++k) {
284 px[k] += this->a(i * NVARS + k) * (pow_x_kx * pow_y_ky - this->c(i));
285 }
286 pow_y_ky *= y;
287 }
288
289 pow_x_kx *= x;
290 }
291
292 return px;
293}

295template <int MAX_DEGREE, int NVARS>
296Cartesian<NVARS> PolyND<MAX_DEGREE, NVARS>::eval_3d(const XYZ &xyz) const {
297 auto [x, y, z] = XYZ((xyz - this->x_center()) / this->reference_length());
298
299 auto d = this->degree();
300
301 Cartesian<NVARS> px(0.0);
302 double pow_x_kx = 1.0;
303 for (int kx = 0; kx <= d; ++kx) {
304 double pow_y_ky = 1.0;
305 for (int ky = 0; ky <= d - kx; ++ky) {
306 double pow_z_kz = 1.0;
307
308 for (int kz = 0; kz <= d - kx - ky; ++kz) {
309 int_t i = idx(kx, ky, kz);
310 for (int_t k = 0; k < NVARS; ++k) {
311 double ak = this->a(i * NVARS + k);
312 px[k] += ak * (pow_x_kx * pow_y_ky * pow_z_kz - this->c(i));
313 }
314
315 pow_z_kz *= z;
316 }
317 pow_y_ky *= y;
318 }
319 pow_x_kx *= x;
320 }
321
322 return px;
323}

325template <int NCOEFFS, int NVARS>
326Cartesian<NVARS> smoothness_indicator(const PolyND<NCOEFFS, NVARS> &p) {
327 auto beta = Cartesian<NVARS>(0.0);
328
329 int_t n = p.dof();
330 for (int_t i = poly_dof(0, p.n_dims()); i < n; ++i) {
331 for (int_t k = 0; k < p.n_vars(); ++k) {
332 beta[k] += zisa::pow<2>(p.a(i * NVARS + k));
333 }
334 }
335
336 return beta;
337}
