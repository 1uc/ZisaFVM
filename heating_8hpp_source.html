18template <class GRC>
19class Heating : public RateOfChange {
20public:
21 Heating(std::shared_ptr<Grid> grid,
22 std::shared_ptr<GRC> grc,
23 std::function<double(const XYZ &x)> heating_rate)
24 : grid(std::move(grid)),
25 grc(std::move(grc)),
26 heating_rate(std::move(heating_rate)) {}
27
28 virtual void compute(AllVariables &tendency,
29 const AllVariables & /* current_state */,
30 double /* t */) const override {
31
32 auto &dudt = tendency.cvars;
33
34 zisa::for_each(cells(*grid), [this, &dudt](int_t i, const Cell &cell) {
35 auto f = [this, i](const XYZ &x) {
36 auto rho
37 = [this, i](const XYZ &x) -> double { return (*grc)(i)(x)[0]; };
38
39 return rho(x) * heating_rate(x);
40 };
41
42 dudt(i, 4) += average(cell, f);
43 });
44 }
45
46 virtual std::string str() const override { return "Constant heating rate."; }
47
48private:
49 std::shared_ptr<Grid> grid;
50 std::shared_ptr<GRC> grc;
51 std::function<double(const XYZ &x)> heating_rate;
52};

54template <class GRC>
55std::shared_ptr<RateOfChange>
56make_heating_source(const std::shared_ptr<Grid> &grid,
57 const std::shared_ptr<GRC> &grc,
58 const InputParameters &params) {
59
60 if (has_key(params, "heating")) {
61 double epsilon = params["heating"]["rate"];
62 double r0 = params["heating"]["lower_boundary"];
63 double r1 = params["heating"]["upper_boundary"];
64
65 auto xi = [r0, r1](const XYZ &x) {
66 double r = zisa::norm(x);
67 return (r0 <= r) && (r <= r1);
68 };
69
70 auto heating_rate
71 = [grid, epsilon, xi](const XYZ &x) { return epsilon * xi(x); };
72
73 return std::make_shared<Heating<GRC>>(grid, grc, heating_rate);
74 }
75
76 return nullptr;
77}
