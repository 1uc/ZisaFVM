36template <class QR, class Domain>
37DenormalizedRule denormalize(const QR &qr_hat, const Domain &domain) {
38 assert(qr_hat.weights.size() == qr_hat.points.size());
39
40 auto n_points = qr_hat.weights.size();
41 auto qr = DenormalizedRule(n_points);
42
43 auto vol = volume(domain);
44 for (int_t i = 0; i < n_points; ++i) {
45 qr.weights[i] = vol * qr_hat.weights[i];
46 qr.points[i] = coord(domain, qr_hat.points[i]);
47 }
48
49 qr.volume = vol;
50 return qr;
51}
