36 class hllc_speeds<IdealGasEOS> {
37 private:
38  using eos_t = IdealGasEOS;
39  using cvars_t = euler_var_t;
40  using xvars_t = euler_var_t::xvars_t;
41 
42 public:
43  static std::tuple<double, double, double> speeds(const eos_t &eos,
44  const cvars_t &uL,
45  const xvars_t &xvarL,
46  const eos_t &,
47  const cvars_t &uR,
48  const xvars_t &xvarR) {
49 
50  RoeAverage roe_average(uL(0), uR(0));
51 
52  auto pL = xvarL.p;
53  auto aL = xvarL.a;
54 
55  auto pR = xvarR.p;
56  auto aR = xvarR.a;
57 
58  double vL = uL(1) / uL(0);
59  double vR = uR(1) / uR(0);
60  double v_tilda = roe_average(vL, vR);
61 
62  double HL = (uL(4) + pL) / uL(0);
63  double HR = (uR(4) + pR) / uR(0);
64  double H_tilda = roe_average(HL, HR);
65 
66  double vroe_square
67  = zisa::pow<2>(roe_average(uL(1) / uL(0), uR(1) / uR(0)))
68  + zisa::pow<2>(roe_average(uL(2) / uL(0), uR(2) / uR(0)))
69  + zisa::pow<2>(roe_average(uL(3) / uL(0), uR(3) / uR(0)));
70 
71  double a_tilda
72  = zisa::sqrt((eos.gamma() - 1.0) * (H_tilda - 0.5 * vroe_square));
73 
74  double sL = zisa::min(vL - aL, v_tilda - a_tilda);
75  double sR = zisa::max(vR + aR, v_tilda + a_tilda);
76  double s_star = (uR(1) * (sR - vR) - uL(1) * (sL - vL) + pL - pR)
77  / (uR(0) * (sR - vR) - uL(0) * (sL - vL));
78 
79  return {sL, s_star, sR};
80  }
81 };

123 template <class EOS>
124 class HLLCBatten {
125 private:
126 public:
127  using cvars_t = euler_var_t;
128  using euler_t = Euler;
129  using eos_t = EOS;
130  using speeds_t = std::tuple<double, double, double>;
131 
132 public:
134  /*
135  * @param euler
136  * @param eos
137  * @param uL
138  * conserved variables 'inside' of the cell, w.r.t the normal
139  * @param uR
140  * conserved variables 'outside' of the cell, w.r.t. the normal.
141  */
142  ANY_DEVICE_INLINE static std::tuple<euler_var_t, speeds_t>
143  flux(const euler_t &euler,
144  const eos_t &eosL,
145  const cvars_t &uL,
146  const eos_t &eosR,
147  const cvars_t &uR) {
148 
149  const auto xvarL = eosL.xvars(uL);
150  const auto xvarR = eosR.xvars(uR);
151 
152  const auto [sL, s_star, sR]
153  = hllc_speeds<eos_t>::speeds(eosL, uL, xvarL, eosR, uR, xvarR);
154 
155  // HLLC flux
156  const cvars_t &uK = (0.0 <= s_star ? uL : uR);
157  const double pK = (0.0 <= s_star ? xvarL.p : xvarR.p);
158  auto nf = euler.flux(uK, pK);
159 
160  if (sL < 0.0 && 0.0 <= sR) {
161  double sK = (0.0 <= s_star ? sL : sR);
162  double vK = (0.0 <= s_star ? uL(1) / uL(0) : uR(1) / uR(0));
163  double cK = (sK - vK) / (sK - s_star);
164 
165  nf(0) += sK * (cK * uK(0) - uK(0));
166  nf(1) += sK * (cK * uK(0) * s_star - uK(1));
167  nf(2) += sK * (cK * uK(2) - uK(2));
168  nf(3) += sK * (cK * uK(3) - uK(3));
169  nf(4)
170  += sK
171  * (cK * (uK(4) + (s_star - vK) * (uK(0) * s_star + pK / (sK - vK)))
172  - uK(4));
173  }
174 
175  return {nf, std::tuple(sL, s_star, sR)};
176  }

178  ANY_DEVICE_INLINE static double tracer_flux(const cvars_t &uL,
179  const cvars_t &uR,
180  double mqL,
181  double mqR,
182  const speeds_t &speeds) {
183 
184  const auto [sL, s_star, sR] = speeds;
185  double mqK = (0.0 <= s_star ? mqL : mqR);
186  double vK = (0.0 <= s_star ? uL(1) / uL(0) : uR(1) / uR(0));
187  double fK = mqK * vK;
188 
189  if (sL < 0.0 && 0.0 < sR) {
190  double sK = (0.0 <= s_star ? sL : sR);
191  double cK = (sK - vK) / (sK - s_star);
192 
193  return fK + sK * (cK * mqK - mqK);
194  }
195 
196  return fK;
197  }
